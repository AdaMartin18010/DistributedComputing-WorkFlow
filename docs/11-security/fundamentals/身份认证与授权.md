# 身份认证与授权

## 概述

身份认证（Authentication）和授权（Authorization）是分布式系统安全的核心组件。认证回答"你是谁"的问题，而授权回答"你能做什么"的问题。在分布式环境中，由于系统组件分散、用户和服务的多样性，设计安全、可扩展的认证授权体系尤为关键。

**JWT实现示例**：

```python
import jwt
import datetime
from typing import Dict, Optional, List
from cryptography.hazmat.primitives import serialization
from cryptography.hazmat.primitives.asymmetric import rsa

class JWTAuthManager:
    def __init__(self, private_key: str, public_key: str, 
                 algorithm: str = "RS256"):
        self.private_key = private_key
        self.public_key = public_key
        self.algorithm = algorithm
        self.issuer = "auth.distributed.system"
        
    def generate_token(self, user_id: str, 
                      roles: List[str],
                      permissions: List[str],
                      expires_hours: int = 24) -> str:
        """
        生成JWT访问令牌
        
        Args:
            user_id: 用户唯一标识
            roles: 用户角色列表
            permissions: 权限列表
            expires_hours: 令牌有效期（小时）
        """
        now = datetime.datetime.utcnow()
        
        payload = {
            # 标准声明
            "sub": user_id,                          # 主题
            "iss": self.issuer,                      # 签发者
            "iat": now,                              # 签发时间
            "exp": now + datetime.timedelta(hours=expires_hours),
            "jti": self._generate_jti(),             # JWT唯一标识
            
            # 自定义声明
            "roles": roles,
            "permissions": permissions,
            "token_type": "access"
        }
        
        token = jwt.encode(
            payload, 
            self.private_key, 
            algorithm=self.algorithm,
            headers={"kid": "key-1"}  # 密钥ID，用于密钥轮换
        )
        
        return token
    
    def generate_refresh_token(self, user_id: str, 
                               expires_days: int = 7) -> str:
        """生成刷新令牌"""
        now = datetime.datetime.utcnow()
        
        payload = {
            "sub": user_id,
            "iss": self.issuer,
            "iat": now,
            "exp": now + datetime.timedelta(days=expires_days),
            "jti": self._generate_jti(),
            "token_type": "refresh"
        }
        
        return jwt.encode(payload, self.private_key, 
                         algorithm=self.algorithm)
    
    def verify_token(self, token: str, 
                    required_audience: Optional[str] = None) -> Dict:
        """
        验证JWT令牌
        
        Raises:
            jwt.ExpiredSignatureError: 令牌已过期
            jwt.InvalidTokenError: 无效令牌
        """
        try:
            payload = jwt.decode(
                token,
                self.public_key,
                algorithms=[self.algorithm],
                issuer=self.issuer,
                audience=required_audience
            )
            
            # 验证令牌类型
            if payload.get("token_type") != "access":
                raise jwt.InvalidTokenError("Invalid token type")
            
            return payload
            
        except jwt.ExpiredSignatureError:
            raise jwt.ExpiredSignatureError("Token has expired")
        except jwt.InvalidSignatureError:
            raise jwt.InvalidTokenError("Invalid token signature")
    
    def _generate_jti(self) -> str:
        """生成JWT唯一标识符"""
        import uuid
        return str(uuid.uuid4())


# 使用示例
if __name__ == "__main__":
    # 生成RSA密钥对（实际应从安全存储加载）
    private_key = rsa.generate_private_key(
        public_exponent=65537,
        key_size=2048
    )
    
    pem_private = private_key.private_bytes(
        encoding=serialization.Encoding.PEM,
        format=serialization.PrivateFormat.PKCS8,
        encryption_algorithm=serialization.NoEncryption()
    ).decode()
    
    pem_public = private_key.public_key().public_bytes(
        encoding=serialization.Encoding.PEM,
        format=serialization.PublicFormat.SubjectPublicKeyInfo
    ).decode()
    
    # 初始化认证管理器
    auth_manager = JWTAuthManager(pem_private, pem_public)
    
    # 生成令牌
    token = auth_manager.generate_token(
        user_id="user123",
        roles=["user", "premium"],
        permissions=["read:profile", "write:profile"],
        expires_hours=1
    )
    
    print(f"Generated JWT: {token[:50]}...")
    
    # 验证令牌
    try:
        payload = auth_manager.verify_token(token)
        print(f"Token valid. User: {payload['sub']}, Roles: {payload['roles']}")
    except jwt.ExpiredSignatureError:
        print("Token expired")
```

#### 2.1.2 令牌刷新机制

```
访问令牌 + 刷新令牌 流程：

┌─────────┐                                   ┌─────────────┐
│  客户端  │                                   │  认证服务器  │
└────┬────┘                                   └──────┬──────┘
     │                                               │
     │  1. 用户名 + 密码                              │
     │ ─────────────────────────────────────────────▶│
     │                                               │
     │  2. Access Token + Refresh Token              │
     │ ◀─────────────────────────────────────────────│
     │                                               │
     │  3. 使用 Access Token 访问 API                │
     │ ──────────────────────▶                       │
     │                         │                     │
     │  4. Token 过期 (401)    │                     │

     │                                               │
     │  5. 使用 Refresh Token 获取新 Token           │
     │ ─────────────────────────────────────────────▶│

     │ ──────────────────────▶                       │

- Refresh Token 有效期更长但只能用于获取新 Token

- Refresh Token 存储需要额外保护（HttpOnly Cookie）
```

**Python mTLS客户端实现**：

```python
import ssl
import socket
import certifi

class MutualTLSClient:
    def __init__(self, 
                 client_cert_path: str,
                 client_key_path: str,
                 ca_cert_path: str = None):
        self.client_cert = client_cert_path
        self.client_key = client_key_path
        self.ca_cert = ca_cert_path or certifi.where()
    
    def create_secure_socket(self) -> ssl.SSLSocket:
        """创建mTLS认证的socket连接"""
        
        # 创建SSL上下文
        context = ssl.SSLContext(ssl.PROTOCOL_TLS_CLIENT)
        context.minimum_version = ssl.TLSVersion.TLSv1_3
        
        # 加载客户端证书和私钥
        context.load_cert_chain(
            certfile=self.client_cert,
            keyfile=self.client_key
        )
        
        # 加载CA证书用于验证服务端
        context.load_verify_locations(self.ca_cert)
        
        # 要求验证服务端证书
        context.verify_mode = ssl.CERT_REQUIRED
        
        # 启用证书吊销检查
        context.verify_flags = ssl.VERIFY_CRL_CHECK_CHAIN
        
        # 创建socket并包装
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        secure_sock = context.wrap_socket(sock, 
                                         server_hostname='api.service.local')
        
        return secure_sock


# 在微服务中使用（以gRPC为例）
import grpc

def create_mtls_channel(target: str) -> grpc.Channel:
    """创建gRPC mTLS通道"""
    
    # 读取证书
    with open('client.crt', 'rb') as f:
        client_cert = f.read()
    with open('client.key', 'rb') as f:
        client_key = f.read()
    with open('ca.crt', 'rb') as f:
        ca_cert = f.read()
    
    # 创建SSL凭证
    credentials = grpc.ssl_channel_credentials(
        root_certificates=ca_cert,
        private_key=client_key,
        certificate_chain=client_cert
    )
    
    # 创建安全通道
    channel = grpc.secure_channel(target, credentials)
    
    return channel
```

### 3.2 ABAC策略引擎实现

```python
from typing import Dict, List, Any, Callable
import json
from dataclasses import dataclass
from enum import Enum

class Effect(Enum):
    ALLOW = "Allow"
    DENY = "Deny"

@dataclass
class Policy:
    """ABAC策略定义"""
    id: str
    effect: Effect
    subjects: List[Dict[str, Any]]  # 主体条件
    resources: List[Dict[str, Any]] # 资源条件
    actions: List[str]              # 允许的操作
    conditions: List[Dict[str, Any]] # 环境条件
    
@dataclass
class AccessRequest:
    """访问请求"""
    subject: Dict[str, Any]   # 请求主体属性
    resource: Dict[str, Any]  # 目标资源属性
    action: str               # 请求操作
    environment: Dict[str, Any] # 环境属性

class ABACEngine:
    """ABAC策略决策引擎"""
    
    def __init__(self):
        self.policies: List[Policy] = []
        self.operators = {
            "eq": lambda a, b: a == b,
            "ne": lambda a, b: a != b,
            "gt": lambda a, b: a > b,
            "gte": lambda a, b: a >= b,
            "lt": lambda a, b: a < b,
            "lte": lambda a, b: a <= b,
            "in": lambda a, b: a in b,
            "contains": lambda a, b: b in a,
            "startswith": lambda a, b: str(a).startswith(str(b)),
            "matches": lambda a, b: self._match_wildcard(a, b)
        }
    
    def add_policy(self, policy: Policy):
        self.policies.append(policy)
    
    def evaluate(self, request: AccessRequest) -> tuple[bool, str]:
        """
        评估访问请求
        
        Returns:
            (是否允许, 决策理由)
        """
        decisions = []
        
        for policy in self.policies:
            if self._policy_applies(policy, request):
                decisions.append(policy)
        
        # 决策合并：Deny优先
        for policy in decisions:
            if policy.effect == Effect.DENY:
                return False, f"Denied by policy {policy.id}"
        
        for policy in decisions:
            if policy.effect == Effect.ALLOW:
                return True, f"Allowed by policy {policy.id}"
        
        return False, "No matching policy found - implicit deny"
    
    def _policy_applies(self, policy: Policy, 
                       request: AccessRequest) -> bool:
        """检查策略是否适用于请求"""
        
        # 检查主体匹配
        if not self._match_attributes(policy.subjects, request.subject):
            return False
        
        # 检查资源匹配
        if not self._match_attributes(policy.resources, request.resource):
            return False
        
        # 检查操作匹配
        if request.action not in policy.actions and "*" not in policy.actions:
            return False
        
        # 检查环境条件
        if not self._match_conditions(policy.conditions, request.environment):
            return False
        
        return True
    
    def _match_attributes(self, conditions: List[Dict], 
                         target: Dict) -> bool:
        """匹配属性条件"""
        for condition in conditions:
            attribute = condition.get("attribute")
            operator = condition.get("operator", "eq")
            value = condition.get("value")
            
            if attribute not in target:
                return False
            
            target_value = target[attribute]
            op_func = self.operators.get(operator)
            
            if not op_func or not op_func(target_value, value):
                return False
        
        return True
    
    def _match_conditions(self, conditions: List[Dict],
                         environment: Dict) -> bool:
        """匹配环境条件"""
        return self._match_attributes(conditions, environment)
    
    def _match_wildcard(self, value: str, pattern: str) -> bool:
        """通配符匹配"""
        import fnmatch
        return fnmatch.fnmatch(str(value), pattern)


# 使用示例
if __name__ == "__main__":
    engine = ABACEngine()
    
    # 定义策略：销售经理工作时间可以访问客户数据
    policy = Policy(
        id="policy-001",
        effect=Effect.ALLOW,
        subjects=[
            {"attribute": "department", "operator": "eq", "value": "sales"},
            {"attribute": "level", "operator": "gte", "value": 3}
        ],
        resources=[
            {"attribute": "type", "operator": "eq", "value": "customer_data"},
            {"attribute": "region", "operator": "eq", "value": "APAC"}
        ],
        actions=["read", "update"],
        conditions=[
            {"attribute": "time_hour", "operator": "gte", "value": 9},
            {"attribute": "time_hour", "operator": "lte", "value": 18},
            {"attribute": "location", "operator": "eq", "value": "office"}
        ]
    )
    
    engine.add_policy(policy)
    
    # 测试请求
    request = AccessRequest(
        subject={"id": "alice", "department": "sales", "level": 4},
        resource={"type": "customer_data", "region": "APAC", "id": "cust123"},
        action="read",
        environment={"time_hour": 14, "location": "office"}
    )
    
    allowed, reason = engine.evaluate(request)
    print(f"Access { 'allowed' if allowed else 'denied' }: {reason}")
```

## 四、OAuth2与OIDC集成

### 4.1 OAuth2授权流程

```
授权码流程（最安全的流程）：

┌─────────┐                                    ┌─────────┐
│   用户   │                                    │  客户端  │
│ (浏览器) │                                    │ (应用)  │
└────┬────┘                                    └────┬────┘
     │                                              │
     │  1. 访问应用                                  │
     │ ────────────────────────────────────────────▶│
     │                                              │
     │  2. 重定向到授权服务器                         │
     │ ◀────────────────────────────────────────────│
     │     response_type=code                       │
     │     client_id=xxx                            │
     │     redirect_uri=xxx                         │
     │     scope=read write                         │
     │     state=random_string                      │
     │                                              │
     │  3. 用户登录并授权                             │
     │ ───────────────────────────────────────────▶┌┴────────┐
     │                                             │授权服务器│
     │  4. 重定向回应用（带授权码）                    └┬────────┘
     │ ◀────────────────────────────────────────────│
     │     code=xxx                                 │
     │     state=random_string                      │
     │                                              │
     │  5. 使用授权码换取令牌                         │
     │ ────────────────────────────────────────────▶│
     │     grant_type=authorization_code            │
     │     code=xxx                                 │
     │     client_secret=xxx                        │
     │                                              │
     │  6. 返回访问令牌和刷新令牌                       │
     │ ◀────────────────────────────────────────────│
     │     access_token=xxx                         │
     │     refresh_token=xxx                        │
     │     expires_in=3600                          │
     │                                              │
     │  7. 使用访问令牌访问资源                        │
     │ ────────────────────────────────────────────▶┌┴────────┐
     │                                             │ 资源服务器│
     │  8. 返回受保护资源                             └┬────────┘
     │ ◀────────────────────────────────────────────│
```

**安全考虑**：
- `state`参数防止CSRF攻击
- PKCE扩展保护移动应用
- 授权码只能使用一次
- 短有效期的访问令牌 + 长有效期的刷新令牌

## 五、最佳实践

### 5.1 令牌安全

```python
# 令牌安全配置
TOKEN_SECURITY_CONFIG = {
    # JWT配置
    "jwt": {
        "algorithm": "RS256",  # 使用非对称加密
        "access_token_ttl": 900,  # 15分钟
        "refresh_token_ttl": 604800,  # 7天
        "issuer": "https://auth.company.com",
        "audience": "https://api.company.com"
    },
    
    # Cookie配置
    "cookie": {
        "http_only": True,  # 防止XSS窃取
        "secure": True,     # 仅HTTPS传输
        "same_site": "Strict",  # CSRF防护
        "max_age": 604800
    },
    
    # 刷新令牌配置
    "refresh_token": {
        "rotation": True,   # 每次使用都生成新令牌
        "reuse_detection": True,  # 检测重用攻击
        "family_binding": True    # 令牌族绑定
    }
}
```

### 5.2 权限设计原则

1. **最小权限原则**：只授予完成任务所需的最小权限
2. **职责分离**：敏感操作需要多个角色共同完成
3. **权限审查**：定期审查和回收不必要的权限
4. **审计日志**：记录所有授权决策和访问行为
5. **失效保护**：认证失败时默认拒绝访问
