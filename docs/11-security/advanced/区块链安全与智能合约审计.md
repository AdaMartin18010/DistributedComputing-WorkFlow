# 区块链安全与智能合约审计

## 概述

区块链技术为分布式系统提供了去中心化、不可篡改的数据存储和价值转移能力。然而，智能合约一旦部署就无法修改的特性，使得安全漏洞的代价极其高昂。本文将系统介绍区块链安全威胁、智能合约常见漏洞及其防御措施，以及审计方法论。

## 二、智能合约漏洞

### 2.1 重入攻击（Reentrancy）

```solidity
// 漏洞合约 - 不安全的提款函数
contract VulnerableBank {
    mapping(address => uint256) public balances;

    function deposit() public payable {
        balances[msg.sender] += msg.value;
    }

    // ❌ 漏洞：先转账后更新余额
    function withdraw() public {
        uint256 amount = balances[msg.sender];

        // 危险：外部调用在状态更新之前
        (bool success, ) = msg.sender.call{value: amount}("");
        require(success, "Transfer failed");

        balances[msg.sender] = 0;  // 更新太晚！
    }
}

// 攻击合约
contract Attacker {
    VulnerableBank public bank;
    uint256 public attackCount;

    constructor(address _bank) {
        bank = VulnerableBank(_bank);
    }

    // 接收ETH时自动调用
    receive() external payable {
        if (attackCount < 10) {
            attackCount++;
            bank.withdraw();  // 递归调用！
        }
    }

    function attack() external payable {
        bank.deposit{value: msg.value}();
        bank.withdraw();
    }
}


// ✅ 修复版本 - Checks-Effects-Interactions模式
contract SecureBank {
    mapping(address => uint256) public balances;
    bool private locked;  // 重入锁

    modifier nonReentrant() {
        require(!locked, "Reentrant call");
        locked = true;
        _;
        locked = false;
    }

    function deposit() public payable {
        balances[msg.sender] += msg.value;
    }

    function withdraw() public nonReentrant {
        uint256 amount = balances[msg.sender];

        // 1. Checks（检查）
        require(amount > 0, "No balance");

        // 2. Effects（效果）- 先更新状态
        balances[msg.sender] = 0;

        // 3. Interactions（交互）- 最后外部调用
        (bool success, ) = msg.sender.call{value: amount}("");
        require(success, "Transfer failed");
    }
}
```

### 2.2 整数溢出与下溢

```solidity
// 漏洞合约 - Solidity < 0.8.0
contract VulnerableToken {
    mapping(address => uint256) public balances;

    // ❌ 漏洞：无溢出检查
    function transfer(address to, uint256 amount) public {
        require(balances[msg.sender] >= amount);
        balances[msg.sender] -= amount;
        balances[to] += amount;  // 可能溢出！
    }

    // 攻击：如果balances[to]接近uint256最大值
    // 再转入少量token会使余额变为很小的数
}

// ✅ 修复版本 1：使用SafeMath
contract SecureTokenV1 {
    using SafeMath for uint256;
    mapping(address => uint256) public balances;

    function transfer(address to, uint256 amount) public {
        balances[msg.sender] = balances[msg.sender].sub(amount);
        balances[to] = balances[to].add(amount);  // 溢出会revert
    }
}

// ✅ 修复版本 2：Solidity >= 0.8.0（内置溢出检查）
contract SecureTokenV2 {
    mapping(address => uint256) public balances;

    function transfer(address to, uint256 amount) public {
        // 0.8.0+ 自动检查溢出，会revert
        balances[msg.sender] -= amount;
        balances[to] += amount;
    }

    // 如需故意溢出，使用unchecked
    function unsafeOperation() public pure returns (uint256) {
        unchecked {
            return uint256(0) - 1;  // 允许溢出
        }
    }
}
```

### 2.4 前端运行（Front-running）

```solidity
// 漏洞场景：DEX交易
contract VulnerableDEX {
    function swap(address token, uint256 amount) public {
        uint256 price = getPrice(token);
        // ❌ 漏洞：交易按gas价格排序，可被抢跑
        // 攻击者看到交易后，支付更高gas抢先交易
        _executeSwap(token, amount, price);
    }
}

// ✅ 修复版本：使用commit-reveal机制
contract SecureDEX {
    struct Commit {
        bytes32 hash;
        uint256 block;
    }

    mapping(address => Commit) public commits;

    // 阶段1：提交（commit）
    function commitSwap(bytes32 hash) public {
        commits[msg.sender] = Commit(hash, block.number);
    }

    // 阶段2：揭示（reveal）
    function revealSwap(
        address token,
        uint256 amount,
        uint256 minReturn,
        bytes32 secret
    ) public {
        Commit memory c = commits[msg.sender];
        require(c.block > 0, "No commit");
        require(block.number > c.block + 10, "Too early");  // 延迟
        require(
            keccak256(abi.encodePacked(token, amount, minReturn, secret)) == c.hash,
            "Invalid reveal"
        );

        // 检查滑点保护
        uint256 price = getPrice(token);
        uint256 expectedReturn = amount * price;
        require(expectedReturn >= minReturn, "Slippage too high");

        _executeSwap(token, amount, expectedReturn);

        delete commits[msg.sender];
    }
}
```

```python
class SlitherAnalyzer:
    """
    Slither静态分析器集成
    """

    def __init__(self, contract_path: str):
        self.contract_path = contract_path

    def analyze(self) -> List[Vulnerability]:
        """运行Slither分析"""
        try:
            result = subprocess.run(
                ['slither', self.contract_path, '--json', '-'],
                capture_output=True,
                text=True,
                timeout=300
            )

            findings = json.loads(result.stdout)
            vulnerabilities = []

            for detector in findings.get('results', []):
                vuln = Vulnerability(
                    title=detector['check'],
                    severity=self._map_severity(detector['impact']),
                    description=detector['description'],
                    location=detector['elements'][0]['source_mapping']['filename'],
                    line_number=detector['elements'][0]['source_mapping']['lines'][0],
                    recommendation=detector.get('recommendation', 'Review and fix'),
                    references=detector.get('references', [])
                )
                vulnerabilities.append(vuln)

            return vulnerabilities

        except Exception as e:
            print(f"Slither analysis failed: {e}")
            return []

    def _map_severity(self, slither_severity: str) -> Severity:
        """映射Slither严重级别"""
        mapping = {
            'High': Severity.HIGH,
            'Medium': Severity.MEDIUM,
            'Low': Severity.LOW,
            'Informational': Severity.INFO
        }
        return mapping.get(slither_severity, Severity.INFO)

class AuditReport:
    """

        self.contract_name = contract_name

{self._generate_summary()}

"""

        # 按严重级别排序

                         Severity.MEDIUM, Severity.LOW, Severity.INFO]

        for severity in severity_order:
            findings = [f for f in self.findings if f.severity == severity]
            if findings:
                report += f"### {severity.value} ({len(findings)})\n\n"
                for i, finding in enumerate(findings, 1):
                    report += self._format_finding(i, finding)

        report += """
## 最佳实践建议

1. 使用最新版本的Solidity编译器
2. 使用OpenZeppelin等经过审计的库
3. 实施全面的单元测试

本审计基于提供的代码快照，不保证发现所有潜在漏洞。
"""

        return report

    def _generate_summary(self) -> str:
        """生成统计摘要"""
        counts = {}
        for f in self.findings:
            counts[f.severity.value] = counts.get(f.severity.value, 0) + 1

        summary = ""
        for sev in ['Critical', 'High', 'Medium', 'Low', 'Informational']:
            count = counts.get(sev, 0)
            summary += f"- **{sev}**: {count}\n"

        return summary

**描述**:
{finding.description}

**修复建议**:
{finding.recommendation}

**参考**:
{chr(10).join('- ' + ref for ref in finding.references)}

---
"""
```

### 4.1 安全开发检查清单

```solidity
// 安全合约模板
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
import "@openzeppelin/contracts/security/Pausable.sol";

/**
 * @title SecureToken
 * @notice 安全ERC20代币示例
 * @dev 包含多种安全最佳实践
 */
contract SecureToken is ERC20, Ownable, ReentrancyGuard, Pausable {

    // ============ 状态变量 ============

    // 使用常量而不是magic numbers
    uint256 public constant MAX_SUPPLY = 100_000_000 * 10**18;
    uint256 public constant MAX_MINT_PER_TX = 1_000_000 * 10**18;

    // 访问控制角色
    mapping(address => bool) public minters;
    mapping(address => bool) public blacklist;

    // 事件
    event MinterAdded(address indexed minter);
    event MinterRemoved(address indexed minter);
    event Blacklisted(address indexed account);
    event Unblacklisted(address indexed account);

    // ============ 修饰符 ============

    modifier onlyMinter() {
        require(minters[msg.sender], "Not a minter");
        _;
    }

    modifier notBlacklisted(address account) {
        require(!blacklist[account], "Account is blacklisted");
        _;
    }

    // ============ 构造函数 ============

    constructor() ERC20("SecureToken", "SEC") {
        _mint(msg.sender, 10_000_000 * 10**18);  // 初始供应
    }

    // ============ 外部函数 ============

    /**
     * @notice 铸造代币（仅铸造者）
     * @param to 接收地址
     * @param amount 数量
     * @dev 有铸币限制和总量限制
     */
    function mint(address to, uint256 amount)
        external
        onlyMinter
        whenNotPaused
        nonReentrant
    {
        require(amount <= MAX_MINT_PER_TX, "Exceeds max mint per tx");
        require(totalSupply() + amount <= MAX_SUPPLY, "Exceeds max supply");
        require(to != address(0), "Cannot mint to zero address");

        _mint(to, amount);
    }

    /**
     * @notice 转账（带黑名单检查）
     */
    function transfer(address to, uint256 amount)
        public
        override
        notBlacklisted(msg.sender)
        notBlacklisted(to)
        whenNotPaused
        returns (bool)
    {
        return super.transfer(to, amount);
    }

    /**
     * @notice 批量转账（gas优化）
     * @dev 批量操作需谨慎处理
     */
    function batchTransfer(
        address[] calldata recipients,
        uint256[] calldata amounts
    ) external whenNotPaused nonReentrant {
        require(recipients.length == amounts.length, "Length mismatch");
        require(recipients.length <= 100, "Batch too large");

        for (uint256 i = 0; i < recipients.length; i++) {
            _transfer(msg.sender, recipients[i], amounts[i]);
        }
    }

    // ============ 管理功能 ============

    function addMinter(address minter) external onlyOwner {
        require(minter != address(0), "Invalid address");
        minters[minter] = true;
        emit MinterAdded(minter);
    }

    function removeMinter(address minter) external onlyOwner {
        minters[minter] = false;
        emit MinterRemoved(minter);
    }

    function addToBlacklist(address account) external onlyOwner {
        blacklist[account] = true;
        emit Blacklisted(account);
    }

    function removeFromBlacklist(address account) external onlyOwner {
        blacklist[account] = false;
        emit Unblacklisted(account);
    }

    function pause() external onlyOwner {
        _pause();
    }

    function unpause() external onlyOwner {
        _unpause();
    }

    // ============ 紧急功能 ============

    /**
     * @notice 紧急提取意外发送的ERC20代币
     */
    function rescueERC20(address token, uint256 amount)
        external
        onlyOwner
    {
        require(token != address(this), "Cannot rescue self");
        IERC20(token).transfer(owner(), amount);
    }
}
```
