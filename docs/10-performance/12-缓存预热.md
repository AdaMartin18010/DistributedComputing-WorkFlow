# 缓存预热

## 核心概念

### 1. 缓存预热的必要性

缓存预热是在服务启动或缓存失效后，提前将热点数据加载到缓存中，避免冷启动导致的性能问题。

**冷启动问题**：
```
┌─────────────────────────────────────────────────────────────┐
│                   缓存冷启动问题                           │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│   场景：新服务上线，缓存为空                                │
│                                                             │
│   请求1：GET /product/10001                                 │
│      ↓ 缓存未命中                                           │
│      查询DB (50ms)                                          │
│      写入缓存                                               │
│      返回结果                                               │
│                                                             │
│   请求2：GET /product/10001                                 │
│      ↓ 缓存命中                                             │
│      返回结果 (<1ms)                                        │
│                                                             │
│   问题：                                                     │
│   1. 启动初期大量请求穿透到数据库                           │
│   2. 数据库压力瞬间增大，可能拖垮服务                       │
│   3. 用户响应时间不稳定                                     │
│   4. 缓存逐渐填充过程中性能波动                             │
│                                                             │
│   解决方案：缓存预热                                         │
│   - 服务启动前加载热点数据                                  │
│   - 流量切入前完成缓存填充                                  │
│   - 确保启动即有高命中率                                    │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

## 实践方法

### 服务启动预热

**1. 启动预热框架**：
```go
package warmup

import (
    "context"
    "sync"
    "time"
)

// WarmUpTask 预热任务接口
type WarmUpTask interface {
    Name() string
    Priority() int  // 优先级，数字越小优先级越高
    Execute(ctx context.Context) error
}

// WarmUpManager 预热管理器
type WarmUpManager struct {
    tasks    []WarmUpTask
    parallel int  // 并发数
}

func NewWarmUpManager(parallel int) *WarmUpManager {
    return &WarmUpManager{
        tasks:    make([]WarmUpTask, 0),
        parallel: parallel,
    }
}

func (m *WarmUpManager) Register(task WarmUpTask) {
    m.tasks = append(m.tasks, task)
}

func (m *WarmUpManager) Run(ctx context.Context) error {
    // 按优先级排序
    sortTasksByPriority(m.tasks)
    
    // 按优先级分组执行
    currentPriority := -1
    var batch []WarmUpTask
    
    for _, task := range m.tasks {
        if task.Priority() != currentPriority {
            // 执行上一批次
            if len(batch) > 0 {
                if err := m.executeBatch(ctx, batch); err != nil {
                    return err
                }
            }
            currentPriority = task.Priority()
            batch = batch[:0]
        }
        batch = append(batch, task)
    }
    
    // 执行最后一批
    if len(batch) > 0 {
        return m.executeBatch(ctx, batch)
    }
    
    return nil
}

func (m *WarmUpManager) executeBatch(ctx context.Context, tasks []WarmUpTask) error {
    var wg sync.WaitGroup
    errCh := make(chan error, len(tasks))
    sem := make(chan struct{}, m.parallel)
    
    for _, task := range tasks {
        wg.Add(1)
        go func(t WarmUpTask) {
            defer wg.Done()
            
            sem <- struct{}{}
            defer func() { <-sem }()
            
            if err := t.Execute(ctx); err != nil {
                errCh <- err
            }
        }(task)
    }
    
    wg.Wait()
    close(errCh)
    
    // 收集错误
    var errs []error
    for err := range errCh {
        errs = append(errs, err)
    }
    
    if len(errs) > 0 {
        return fmt.Errorf("warmup errors: %v", errs)
    }
    
    return nil
}
```

**2. 具体预热任务实现**：
```go
// 商品缓存预热
type ProductWarmUpTask struct {
    productService ProductService
    cache          Cache
    topN           int
}

func (t *ProductWarmUpTask) Name() string {
    return "product_cache_warmup"
}

func (t *ProductWarmUpTask) Priority() int {
    return 1  // 高优先级
}

func (t *ProductWarmUpTask) Execute(ctx context.Context) error {
    // 1. 获取热门商品ID
    productIDs, err := t.productService.GetTopSellingProducts(ctx, t.topN)
    if err != nil {
        return err
    }
    
    // 2. 批量加载到缓存
    batchSize := 100
    for i := 0; i < len(productIDs); i += batchSize {
        end := i + batchSize
        if end > len(productIDs) {
            end = len(productIDs)
        }
        batch := productIDs[i:end]
        
        products, err := t.productService.BatchGetProducts(ctx, batch)
        if err != nil {
            log.Printf("Failed to load products batch: %v", err)
            continue
        }
        
        // 写入缓存
        for _, p := range products {
            key := fmt.Sprintf("product:%d", p.ID)
            if err := t.cache.Set(ctx, key, p, 30*time.Minute); err != nil {
                log.Printf("Failed to cache product %d: %v", p.ID, err)
            }
        }
        
        // 避免对数据库造成压力
        time.Sleep(100 * time.Millisecond)
    }
    
    log.Printf("Product cache warmup completed: %d products", len(productIDs))
    return nil
}

// 用户会话预热
type SessionWarmUpTask struct {
    sessionService SessionService
    cache          Cache
}

func (t *SessionWarmUpTask) Execute(ctx context.Context) error {
    // 仅预热活跃用户会话
    activeSessions, err := t.sessionService.GetRecentActiveSessions(ctx, 24*time.Hour)
    if err != nil {
        return err
    }
    
    for _, session := range activeSessions {
        key := fmt.Sprintf("session:%s", session.ID)
        t.cache.Set(ctx, key, session, session.TTL)
    }
    
    log.Printf("Session cache warmup completed: %d sessions", len(activeSessions))
    return nil
}
```

**3. 健康检查与流量控制**：
```go
// 服务启动流程
func main() {
    // 1. 初始化基础设施
    db := initDB()
    cache := initCache()
    
    // 2. 执行预热
    warmupManager := warmup.NewWarmUpManager(10)
    warmupManager.Register(&ProductWarmUpTask{...})
    warmupManager.Register(&SessionWarmUpTask{...})
    warmupManager.Register(&ConfigWarmUpTask{...})
    
    ctx, cancel := context.WithTimeout(context.Background(), 5*time.Minute)
    defer cancel()
    
    if err := warmupManager.Run(ctx); err != nil {
        log.Fatalf("Warmup failed: %v", err)
    }
    
    // 3. 预热完成，启动服务
    server := initServer()
    
    // 4. 健康检查通过后再注册到服务发现
    if err := healthCheck(); err != nil {
        log.Fatalf("Health check failed: %v", err)
    }
    
    registerToDiscovery()
    server.Start()
}

// 健康检查
func healthCheck() error {
    // 检查缓存命中率
    hitRate := cache.GetHitRate()
    if hitRate < 0.5 {
        return fmt.Errorf("cache hit rate too low: %.2f", hitRate)
    }
    
    // 检查关键数据是否存在
    criticalKeys := []string{"config:system", "config:rate_limit"}
    for _, key := range criticalKeys {
        if _, ok := cache.Get(key); !ok {
            return fmt.Errorf("critical key not found: %s", key)
        }
    }
    
    return nil
}
```

type ScheduledWarmUp struct {
    scheduler *cron.Cron
    manager   *WarmUpManager
}

func NewScheduledWarmUp() *ScheduledWarmUp {
    s := &ScheduledWarmUp{
        scheduler: cron.New(cron.WithSeconds()),
        manager:   NewWarmUpManager(10),
    }
    
    // 注册定时任务
    // 每天凌晨3点预热
    s.scheduler.AddFunc("0 0 3 * * *", s.dailyWarmUp)
    
    // 每小时预热热点数据
    s.scheduler.AddFunc("0 0 * * * *", s.hotDataWarmUp)
    
    return s
}

func (s *ScheduledWarmUp) dailyWarmUp() {
    ctx := context.Background()
    
    // 预热今日推荐商品
    s.manager.Register(&DailyRecommendWarmUpTask{})
    
    // 预热今日活动
    s.manager.Register(&TodayActivityWarmUpTask{})
    
    if err := s.manager.Run(ctx); err != nil {
        log.Printf("Daily warmup failed: %v", err)
        alert.Send("Daily warmup failed", err)
    }
}

func (s *ScheduledWarmUp) hotDataWarmUp() {
    // 基于最近1小时访问日志，预热热点数据

    }

### 渐进式预热

**1. 灰度预热**：
```go
// 渐进式流量切换
type GradualWarmUp struct {
    cache        Cache
    trafficRate  float64  // 当前流量比例
    targetRate   float64  // 目标流量比例
    step         float64  // 每步增加比例
    interval     time.Duration
}

func (g *GradualWarmUp) Start() {
    ticker := time.NewTicker(g.interval)
    defer ticker.Stop()
    
    for range ticker.C {
        if g.trafficRate >= g.targetRate {
            break
        }
        
        g.trafficRate += g.step
        if g.trafficRate > g.targetRate {
            g.trafficRate = g.targetRate
        }
        
        // 调整负载均衡权重
        updateTrafficWeight(g.trafficRate)
        
        // 监控缓存命中率
        hitRate := g.cache.GetHitRate()
        if hitRate < 0.8 {
            // 命中率不足，暂停增加流量
            log.Printf("Hit rate too low (%.2f), pausing traffic increase", hitRate)
            continue
        }
        
        log.Printf("Traffic rate increased to %.2f%%, hit rate: %.2f%%", 
            g.trafficRate*100, hitRate*100)
    }
}

// 使用
gradual := &GradualWarmUp{
    cache:       cache,
    trafficRate: 0,
    targetRate:  1.0,
    step:        0.1,
    interval:    1 * time.Minute,
}
go gradual.Start()
```

// 预热指标
type WarmUpMetrics struct {
    taskDuration  *prometheus.HistogramVec
    taskSuccess   *prometheus.CounterVec
    taskFailure   *prometheus.CounterVec
    keysWarmed    prometheus.Counter
}

func (m *WarmUpMetrics) RecordTask(taskName string, duration time.Duration, err error) {
    m.taskDuration.WithLabelValues(taskName).Observe(duration.Seconds())
    if err != nil {
        m.taskFailure.WithLabelValues(taskName).Inc()
    } else {
        m.taskSuccess.WithLabelValues(taskName).Inc()
    }
}
