# 容量预测模型

## 核心概念

### 1. 容量预测的必要性

容量预测通过分析历史数据和业务趋势，提前规划资源，避免系统过载或资源浪费。

**预测场景**：

```
┌─────────────────────────────────────────────────────────────┐
│                   容量预测应用场景                         │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│   1. 业务增长预测                                            │
│      - 日活用户增长 → 系统QPS增长 → 服务器需求             │
│      - 订单量增长 → 数据库存储量增长 → 存储扩容            │
│                                                             │
│   2. 大促容量规划                                            │
│      - 预测双11峰值流量                                      │
│      - 提前3个月开始扩容                                     │
│      - 资源采购周期匹配                                      │
│                                                             │
│   3. 成本优化                                                │
│      - 识别低利用率资源                                      │
│      - 预测业务低谷，缩容节省成本                            │
│      - 混合云弹性伸缩规划                                    │
│                                                             │
│   4. 风险管理                                                │
│      - 提前发现容量缺口                                      │
│      - 制定扩容计划                                          │
│      - 避免被动应急                                          │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 2. 预测方法分类

| 方法 | 描述 | 优点 | 缺点 | 适用场景 |
|------|------|------|------|----------|
| 线性回归 | 假设线性增长 | 简单 | 非线性场景不准 | 稳定增长 |
| 时间序列 | ARIMA/Prophet | 考虑趋势和周期 | 需要历史数据 | 周期性业务 |
| 机器学习 | LSTM/XGBoost | 精度高 | 需要大量数据 | 复杂场景 |
| 业务驱动 | 基于业务指标 | 贴合业务 | 需要业务输入 | 大促规划 |

## 实践方法

### 时间序列预测

**1. Prophet预测模型**：

```python
from prophet import Prophet
import pandas as pd

class CapacityPredictor:
    """容量预测器"""

    def __init__(self):
        self.model = Prophet(
            yearly_seasonality=True,
            weekly_seasonality=True,
            daily_seasonality=False,
            changepoint_prior_scale=0.05
        )

        # 添加节假日效应
        self.model.add_country_holidays(country_name='CN')

    def fit(self, historical_data):
        """训练模型"""
        # 数据格式：ds(日期), y(指标值)
        df = pd.DataFrame(historical_data)
        self.model.fit(df)

    def predict(self, periods=90):
        """预测未来"""
        future = self.model.make_future_dataframe(periods=periods)
        forecast = self.model.predict(future)
        return forecast[['ds', 'yhat', 'yhat_lower', 'yhat_upper']]

    def predict_capacity_needs(self, current_capacity, target_utilization=0.7):
        """预测容量需求"""
        forecast = self.predict(periods=180)  # 预测6个月

        needs = []
        for _, row in forecast.iterrows():
            predicted_qps = row['yhat']
            required_capacity = predicted_qps / target_utilization

            needs.append({
                'date': row['ds'],
                'predicted_qps': predicted_qps,
                'required_capacity': required_capacity,
                'scale_factor': required_capacity / current_capacity
            })

        return pd.DataFrame(needs)

# 使用示例
predictor = CapacityPredictor()

# 历史数据
history = [
    {'ds': '2024-01-01', 'y': 10000},
    {'ds': '2024-01-02', 'y': 10500},
    # ... 更多数据
]

predictor.fit(history)
needs = predictor.predict_capacity_needs(current_capacity=50000)

# 输出预警
alerts = needs[needs['scale_factor'] > 1.2]  # 超过20%容量缺口预警
```

**2. 机器学习预测**：

```python
import xgboost as xgb
from sklearn.model_selection import train_test_split

class MLCapacityPredictor:
    """机器学习容量预测"""

    def __init__(self):
        self.model = xgb.XGBRegressor(
            n_estimators=100,
            max_depth=5,
            learning_rate=0.1
        )

    def prepare_features(self, data):
        """特征工程"""
        features = pd.DataFrame()

        # 时间特征
        features['day_of_week'] = pd.to_datetime(data['date']).dt.dayofweek
        features['month'] = pd.to_datetime(data['date']).dt.month
        features['is_weekend'] = features['day_of_week'].isin([5, 6]).astype(int)

        # 滞后特征
        features['qps_lag_1d'] = data['qps'].shift(1)
        features['qps_lag_7d'] = data['qps'].shift(7)
        features['qps_rolling_mean_7d'] = data['qps'].rolling(7).mean()

        # 业务特征
        features['dau'] = data['dau']
        features['marketing_activity'] = data['marketing_activity']

        return features.dropna()

    def train(self, data):
        """训练模型"""
        X = self.prepare_features(data)
        y = data['qps'][X.index]

        X_train, X_test, y_train, y_test = train_test_split(
            X, y, test_size=0.2, shuffle=False
        )

        self.model.fit(X_train, y_train)

        # 评估
        score = self.model.score(X_test, y_test)
        print(f"Model R² score: {score}")

    def predict(self, future_data):
        """预测"""
        X = self.prepare_features(future_data)
        return self.model.predict(X)
```

### 业务驱动预测

```python
class BusinessDrivenPredictor:
    """业务驱动容量预测"""

    def predict_big_sale(self, business_plan):
        """大促容量预测"""

        # 1. 业务输入
        target_gmv = business_plan['target_gmv']  # 目标GMV
        avg_order_value = business_plan['avg_order_value']  # 客单价
        peak_ratio = business_plan['peak_ratio']  # 峰值占比

        # 2. 计算订单量
        total_orders = target_gmv / avg_order_value
        peak_orders = total_orders * peak_ratio

        # 3. 计算峰值QPS
        peak_hours = 2  # 峰值持续2小时
        peak_qps = peak_orders / (peak_hours * 3600)

        # 4. 系统放大系数
        page_views_per_order = 50  # 每订单浏览50次
        api_calls_per_order = 10   # 每订单10次API调用

        total_qps = peak_qps * (page_views_per_order + api_calls_per_order)

        # 5. 计算资源需求
        qps_per_instance = 1000
        redundancy_factor = 1.3

        required_instances = int(
            total_qps / qps_per_instance * redundancy_factor
        )

        return {
            'peak_qps': total_qps,
            'required_instances': required_instances,
            'peak_orders_per_second': peak_qps,
            'resource_breakdown': {
                'app_servers': required_instances,
                'db_connections': required_instances * 20,
                'cache_memory': self.estimate_cache(peak_qps),
                'mq_throughput': peak_qps
            }
        }

    def estimate_cache(self, qps):
        """估算缓存容量"""
        # 热点数据缓存
        hot_data_ratio = 0.2  # 20%热点数据
        data_per_request = 10 * 1024  # 10KB/请求
        cache_hit_rate = 0.9

        cache_size = qps * data_per_request / cache_hit_rate * hot_data_ratio
        return cache_size
```
