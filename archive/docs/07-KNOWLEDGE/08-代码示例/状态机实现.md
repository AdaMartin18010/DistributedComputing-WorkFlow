## 状态机基础概念

### 什么是状态机

状态机（State Machine）是一种数学计算模型，用于表示有限个状态以及在这些状态之间的转移和动作。

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                          状态机核心组件图                                     │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                             │
│   ┌──────────┐         事件/条件           ┌──────────┐                      │
│   │          │ ──────────────────────────▶ │          │                      │
│   │  状态 A  │        转换 (Transition)     │  状态 B  │                      │
│   │          │                             │          │                      │
│   └──────────┘                             └──────────┘                      │
│        │                                        │                           │
│        │ 进入动作 (Entry Action)                 │ 进入动作                   │
│        ▼                                        ▼                           │
│   ┌──────────┐                             ┌──────────┐                      │
│   │ 执行逻辑 │                             │ 执行逻辑 │                      │
│   └──────────┘                             └──────────┘                      │
│                                                                             │
│   核心要素:                                                                 │
│   • 状态 (State): 系统所处的特定条件或阶段                                     │
│   • 事件 (Event): 触发状态转换的信号                                           │
│   • 转换 (Transition): 从一个状态到另一个状态的规则                             │
│   • 动作 (Action): 状态转换时执行的操作                                         │
│   • 守卫条件 (Guard): 决定转换是否可以执行的条件                                 │
│                                                                             │
└─────────────────────────────────────────────────────────────────────────────┘
```

### 状态机类型

1. **有限状态机 (FSM)**: 最简单形式，每个状态有确定性的转移
2. **层次状态机**: 支持状态嵌套，子状态继承父状态行为
3. **并行状态机**: 支持多个状态同时活跃
4. **工作流状态机**: 结合持久化和长时间运行的特性

### 1. 核心类型定义

```go
// fsm/types.go
package fsm

import (
    "context"
    "fmt"
    "time"
)

// State 表示状态
type State string

// Event 表示事件
type Event string

// Transition 表示状态转换
type Transition struct {
    From      State
    To        State
    Event     Event
    Guard     GuardFunc      // 守卫条件
    Before    ActionFunc     // 转换前动作
    After     ActionFunc     // 转换后动作
}

// GuardFunc 守卫条件函数
// 返回 true 表示允许转换
// 返回 error 表示不允许转换，并给出原因
type GuardFunc func(ctx context.Context, from, to State, event Event, args map[string]interface{}) error

// ActionFunc 动作函数
type ActionFunc func(ctx context.Context, from, to State, event Event, args map[string]interface{}) error

// FSMConfig 状态机配置
type FSMConfig struct {
    Name          string
    InitialState  State
    States        []StateDefinition
    Transitions   []Transition
    FinalStates   []State           // 终止状态
    AutoPersist   bool              // 自动持久化
    PersistInterval time.Duration   // 持久化间隔
}

// StateDefinition 状态定义
type StateDefinition struct {
    Name      State
    OnEntry   ActionFunc   // 进入状态时执行
    OnExit    ActionFunc   // 离开状态时执行
    OnStay    ActionFunc   // 保持在状态时周期性执行
    IsFinal   bool         // 是否是终止状态
}

// FSMInstance 状态机实例
type FSMInstance struct {
    ID          string
    FSMType     string
    CurrentState State
    PreviousState State
    Data        map[string]interface{}  // 业务数据
    History     []TransitionRecord      // 转换历史
    CreatedAt   time.Time
    UpdatedAt   time.Time
    Version     int                     // 乐观锁版本
}

// TransitionRecord 转换记录
type TransitionRecord struct {
    From      State
    To        State
    Event     Event
    Timestamp time.Time
    Args      map[string]interface{}
    Success   bool
    Error     string
}

// FSM 状态机定义
type FSM struct {
    config      FSMConfig
    states      map[State]StateDefinition
    transitions map[State]map[Event]Transition
    persistence Persistence
}

// NewFSM 创建新的状态机
func NewFSM(config FSMConfig, persistence Persistence) (*FSM, error) {
    fsm := &FSM{
        config:      config,
        states:      make(map[State]StateDefinition),
        transitions: make(map[State]map[Event]Transition),
        persistence: persistence,
    }

    // 注册状态
    for _, state := range config.States {
        fsm.states[state.Name] = state
    }

    // 注册转换
    for _, trans := range config.Transitions {
        if _, ok := fsm.transitions[trans.From]; !ok {
            fsm.transitions[trans.From] = make(map[Event]Transition)
        }
        fsm.transitions[trans.From][trans.Event] = trans
    }

    return fsm, nil
}

// CreateInstance 创建状态机实例
func (f *FSM) CreateInstance(id string, initialData map[string]interface{}) (*FSMInstance, error) {
    instance := &FSMInstance{
        ID:           id,
        FSMType:      f.config.Name,
        CurrentState: f.config.InitialState,
        Data:         initialData,
        History:      make([]TransitionRecord, 0),
        CreatedAt:    time.Now(),
        UpdatedAt:    time.Now(),
        Version:      1,
    }

    // 执行初始状态的进入动作
    if state, ok := f.states[f.config.InitialState]; ok && state.OnEntry != nil {
        ctx := context.Background()
        if err := state.OnEntry(ctx, "", f.config.InitialState, "", instance.Data); err != nil {
            return nil, fmt.Errorf("initial state entry action failed: %w", err)
        }
    }

    // 持久化
    if f.persistence != nil {
        if err := f.persistence.Save(context.Background(), instance); err != nil {
            return nil, err
        }
    }

    return instance, nil
}
```

### 2. 状态转换实现

```go
// fsm/transition.go
package fsm

import (
    "context"
    "fmt"
    "time"
)

// Trigger 触发状态转换
func (f *FSM) Trigger(ctx context.Context, instance *FSMInstance, event Event, args map[string]interface{}) error {
    // 检查当前状态是否有效
    currentTrans, ok := f.transitions[instance.CurrentState]
    if !ok {
        return fmt.Errorf("no transitions defined for state %s", instance.CurrentState)
    }

    // 查找对应的转换规则
    transition, ok := currentTrans[event]
    if !ok {
        return fmt.Errorf("event %s not allowed in state %s", event, instance.CurrentState)
    }

    // 检查是否是终止状态
    if state, ok := f.states[instance.CurrentState]; ok && state.IsFinal {
        return fmt.Errorf("cannot transition from final state %s", instance.CurrentState)
    }

    record := TransitionRecord{
        From:      instance.CurrentState,
        To:        transition.To,
        Event:     event,
        Timestamp: time.Now(),
        Args:      args,
    }

    // 执行守卫条件检查
    if transition.Guard != nil {
        if err := transition.Guard(ctx, instance.CurrentState, transition.To, event, args); err != nil {
            record.Success = false
            record.Error = err.Error()
            instance.History = append(instance.History, record)
            return fmt.Errorf("guard condition failed: %w", err)
        }
    }

    // 执行转换前动作
    if transition.Before != nil {
        if err := transition.Before(ctx, instance.CurrentState, transition.To, event, args); err != nil {
            record.Success = false
            record.Error = err.Error()
            instance.History = append(instance.History, record)
            return fmt.Errorf("before action failed: %w", err)
        }
    }

    // 执行当前状态的退出动作
    if currentState, ok := f.states[instance.CurrentState]; ok && currentState.OnExit != nil {
        if err := currentState.OnExit(ctx, instance.CurrentState, transition.To, event, args); err != nil {
            record.Success = false
            record.Error = err.Error()
            instance.History = append(instance.History, record)
            return fmt.Errorf("exit action failed: %w", err)
        }
    }

    // 执行状态转换
    previousState := instance.CurrentState
    instance.PreviousState = previousState
    instance.CurrentState = transition.To
    instance.UpdatedAt = time.Now()
    instance.Version++

    // 合并 args 到实例数据
    for k, v := range args {
        instance.Data[k] = v
    }

    // 执行目标状态的进入动作
    if targetState, ok := f.states[transition.To]; ok && targetState.OnEntry != nil {
        if err := targetState.OnEntry(ctx, previousState, transition.To, event, instance.Data); err != nil {
            // 回滚状态
            instance.CurrentState = previousState
            record.Success = false
            record.Error = err.Error()
            instance.History = append(instance.History, record)
            return fmt.Errorf("entry action failed: %w", err)
        }
    }

    // 执行转换后动作
    if transition.After != nil {
        if err := transition.After(ctx, previousState, transition.To, event, args); err != nil {
            // 注意：这里不自动回滚，因为状态已改变
            // 实际应用中可能需要人工介入或补偿逻辑
            record.Success = false
            record.Error = err.Error()
            instance.History = append(instance.History, record)
            return fmt.Errorf("after action failed: %w", err)
        }
    }

    record.Success = true
    instance.History = append(instance.History, record)

    // 持久化
    if f.persistence != nil {
        if err := f.persistence.Save(ctx, instance); err != nil {
            return fmt.Errorf("persistence failed: %w", err)
        }
    }

    return nil
}

// CanTrigger 检查是否可以触发指定事件
func (f *FSM) CanTrigger(instance *FSMInstance, event Event) bool {
    currentTrans, ok := f.transitions[instance.CurrentState]
    if !ok {
        return false
    }

    _, ok = currentTrans[event]
    return ok
}

// GetAvailableEvents 获取当前状态下可用的事件列表
func (f *FSM) GetAvailableEvents(instance *FSMInstance) []Event {
    currentTrans, ok := f.transitions[instance.CurrentState]
    if !ok {
        return []Event{}
    }

    events := make([]Event, 0, len(currentTrans))
    for event := range currentTrans {
        events = append(events, event)
    }
    return events
}

// IsInFinalState 检查是否处于终止状态
func (f *FSM) IsInFinalState(instance *FSMInstance) bool {
    if state, ok := f.states[instance.CurrentState]; ok {
        return state.IsFinal
    }
    return false
}
```

### 3. 持久化接口与实现

```go
// fsm/persistence.go
package fsm

import (
    "context"
    "encoding/json"
    "fmt"
    "sync"
    "time"
)

// Persistence 持久化接口
type Persistence interface {
    Save(ctx context.Context, instance *FSMInstance) error
    Load(ctx context.Context, id string) (*FSMInstance, error)
    Update(ctx context.Context, instance *FSMInstance) error
    Delete(ctx context.Context, id string) error
    List(ctx context.Context, fsmType string, state State) ([]*FSMInstance, error)
}

// MemoryPersistence 内存持久化实现（用于测试）
type MemoryPersistence struct {
    mu        sync.RWMutex
    instances map[string]*FSMInstance
}

func NewMemoryPersistence() *MemoryPersistence {
    return &MemoryPersistence{
        instances: make(map[string]*FSMInstance),
    }
}

func (m *MemoryPersistence) Save(ctx context.Context, instance *FSMInstance) error {
    m.mu.Lock()
    defer m.mu.Unlock()

    // 深拷贝避免外部修改
    copy := deepCopyInstance(instance)
    m.instances[instance.ID] = copy
    return nil
}

func (m *MemoryPersistence) Load(ctx context.Context, id string) (*FSMInstance, error) {
    m.mu.RLock()
    defer m.mu.RUnlock()

    instance, ok := m.instances[id]
    if !ok {
        return nil, fmt.Errorf("instance not found: %s", id)
    }

    return deepCopyInstance(instance), nil
}

func (m *MemoryPersistence) Update(ctx context.Context, instance *FSMInstance) error {
    return m.Save(ctx, instance)
}

func (m *MemoryPersistence) Delete(ctx context.Context, id string) error {
    m.mu.Lock()
    defer m.mu.Unlock()

    delete(m.instances, id)
    return nil
}

func (m *MemoryPersistence) List(ctx context.Context, fsmType string, state State) ([]*FSMInstance, error) {
    m.mu.RLock()
    defer m.mu.RUnlock()

    var result []*FSMInstance
    for _, instance := range m.instances {
        if instance.FSMType == fsmType && (state == "" || instance.CurrentState == state) {
            result = append(result, deepCopyInstance(instance))
        }
    }
    return result, nil
}

// deepCopyInstance 深拷贝实例
func deepCopyInstance(src *FSMInstance) *FSMInstance {
    data, _ := json.Marshal(src)
    var dst FSMInstance
    json.Unmarshal(data, &dst)
    return &dst
}
```

```go
// storage/postgres.go
package storage

import (
    "context"
    "database/sql"
    "encoding/json"
    "fmt"
    "time"

    _ "github.com/lib/pq"
    "statemachine-go/fsm"
)

// PostgresPersistence PostgreSQL 持久化实现
type PostgresPersistence struct {
    db *sql.DB
}

// NewPostgresPersistence 创建 PostgreSQL 持久化实例
func NewPostgresPersistence(connStr string) (*PostgresPersistence, error) {
    db, err := sql.Open("postgres", connStr)
    if err != nil {
        return nil, err
    }

    // 测试连接
    if err := db.Ping(); err != nil {
        return nil, err
    }

    p := &PostgresPersistence{db: db}

    // 初始化表
    if err := p.initSchema(); err != nil {
        return nil, err
    }

    return p, nil
}

// initSchema 初始化数据库表
func (p *PostgresPersistence) initSchema() error {
    schema := `
    CREATE TABLE IF NOT EXISTS fsm_instances (
        id VARCHAR(255) PRIMARY KEY,
        fsm_type VARCHAR(255) NOT NULL,
        current_state VARCHAR(255) NOT NULL,
        previous_state VARCHAR(255),
        data JSONB,
        history JSONB DEFAULT '[]',
        created_at TIMESTAMP NOT NULL,
        updated_at TIMESTAMP NOT NULL,
        version INTEGER NOT NULL DEFAULT 1
    );

    CREATE INDEX IF NOT EXISTS idx_fsm_type ON fsm_instances(fsm_type);
    CREATE INDEX IF NOT EXISTS idx_current_state ON fsm_instances(current_state);
    CREATE INDEX IF NOT EXISTS idx_updated_at ON fsm_instances(updated_at);
    `

    _, err := p.db.Exec(schema)
    return err
}

func (p *PostgresPersistence) Save(ctx context.Context, instance *fsm.FSMInstance) error {
    dataJSON, _ := json.Marshal(instance.Data)
    historyJSON, _ := json.Marshal(instance.History)

    query := `
        INSERT INTO fsm_instances
        (id, fsm_type, current_state, previous_state, data, history, created_at, updated_at, version)
        VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9)
        ON CONFLICT (id) DO UPDATE SET
            current_state = EXCLUDED.current_state,
            previous_state = EXCLUDED.previous_state,
            data = EXCLUDED.data,
            history = EXCLUDED.history,
            updated_at = EXCLUDED.updated_at,
            version = EXCLUDED.version
    `

    _, err := p.db.ExecContext(ctx, query,
        instance.ID,
        instance.FSMType,
        instance.CurrentState,
        instance.PreviousState,
        dataJSON,
        historyJSON,
        instance.CreatedAt,
        instance.UpdatedAt,
        instance.Version,
    )

    return err
}

func (p *PostgresPersistence) Load(ctx context.Context, id string) (*fsm.FSMInstance, error) {
    query := `
        SELECT id, fsm_type, current_state, previous_state, data, history,
               created_at, updated_at, version
        FROM fsm_instances WHERE id = $1
    `

    row := p.db.QueryRowContext(ctx, query, id)

    instance := &fsm.FSMInstance{}
    var dataJSON, historyJSON []byte

    err := row.Scan(
        &instance.ID,
        &instance.FSMType,
        &instance.CurrentState,
        &instance.PreviousState,
        &dataJSON,
        &historyJSON,
        &instance.CreatedAt,
        &instance.UpdatedAt,
        &instance.Version,
    )

    if err == sql.ErrNoRows {
        return nil, fmt.Errorf("instance not found: %s", id)
    }
    if err != nil {
        return nil, err
    }

    // 解析 JSON
    json.Unmarshal(dataJSON, &instance.Data)
    json.Unmarshal(historyJSON, &instance.History)

    return instance, nil
}

func (p *PostgresPersistence) Update(ctx context.Context, instance *fsm.FSMInstance) error {
    // 乐观锁检查
    query := `
        UPDATE fsm_instances SET
            current_state = $1,
            previous_state = $2,
            data = $3,
            history = $4,
            updated_at = $5,
            version = version + 1
        WHERE id = $6 AND version = $7
    `

    dataJSON, _ := json.Marshal(instance.Data)
    historyJSON, _ := json.Marshal(instance.History)

    result, err := p.db.ExecContext(ctx, query,
        instance.CurrentState,
        instance.PreviousState,
        dataJSON,
        historyJSON,
        time.Now(),
        instance.ID,
        instance.Version,
    )

    if err != nil {
        return err
    }

    rows, _ := result.RowsAffected()
    if rows == 0 {
        return fmt.Errorf("optimistic lock failed: instance %s was modified", instance.ID)
    }

    instance.Version++
    return nil
}

func (p *PostgresPersistence) Delete(ctx context.Context, id string) error {
    _, err := p.db.ExecContext(ctx, "DELETE FROM fsm_instances WHERE id = $1", id)
    return err
}

func (p *PostgresPersistence) List(ctx context.Context, fsmType string, state fsm.State) ([]*fsm.FSMInstance, error) {
    query := `
        SELECT id, fsm_type, current_state, previous_state, data, history,
               created_at, updated_at, version
        FROM fsm_instances
        WHERE fsm_type = $1
    `
    args := []interface{}{fsmType}

    if state != "" {
        query += " AND current_state = $2"
        args = append(args, state)
    }

    rows, err := p.db.QueryContext(ctx, query, args...)
    if err != nil {
        return nil, err
    }
    defer rows.Close()

    var instances []*fsm.FSMInstance
    for rows.Next() {
        instance := &fsm.FSMInstance{}
        var dataJSON, historyJSON []byte

        err := rows.Scan(
            &instance.ID,
            &instance.FSMType,
            &instance.CurrentState,
            &instance.PreviousState,
            &dataJSON,
            &historyJSON,
            &instance.CreatedAt,
            &instance.UpdatedAt,
            &instance.Version,
        )
        if err != nil {
            return nil, err
        }

        json.Unmarshal(dataJSON, &instance.Data)
        json.Unmarshal(historyJSON, &instance.History)

        instances = append(instances, instance)
    }

    return instances, nil
}

func (p *PostgresPersistence) Close() error {
    return p.db.Close()
}
```

### 4. 订单状态机示例

```go
// examples/order_fsm.go
package main

import (
    "context"
    "fmt"
    "log"
    "time"

    "statemachine-go/fsm"
    "statemachine-go/storage"
)

// 订单状态
const (
    OrderStateCreated     fsm.State = "CREATED"
    OrderStatePending     fsm.State = "PENDING_PAYMENT"
    OrderStatePaid        fsm.State = "PAID"
    OrderStatePreparing   fsm.State = "PREPARING"
    OrderStateShipped     fsm.State = "SHIPPED"
    OrderStateDelivered   fsm.State = "DELIVERED"
    OrderStateCancelled   fsm.State = "CANCELLED"
    OrderStateRefunded    fsm.State = "REFUNDED"
)

// 订单事件
const (
    EventSubmit     fsm.Event = "SUBMIT"
    EventPay        fsm.Event = "PAY"
    EventPrepare    fsm.Event = "PREPARE"
    EventShip       fsm.Event = "SHIP"
    EventDeliver    fsm.Event = "DELIVER"
    EventCancel     fsm.Event = "CANCEL"
    EventRefund     fsm.Event = "REFUND"
    EventTimeout    fsm.Event = "TIMEOUT"
)

// CreateOrderFSM 创建订单状态机
func CreateOrderFSM(persistence fsm.Persistence) (*fsm.FSM, error) {
    config := fsm.FSMConfig{
        Name:         "OrderFSM",
        InitialState: OrderStateCreated,
        FinalStates:  []fsm.State{OrderStateDelivered, OrderStateCancelled, OrderStateRefunded},
    }

    // 定义状态
    config.States = []fsm.StateDefinition{
        {
            Name: OrderStateCreated,
            OnEntry: func(ctx context.Context, from, to fsm.State, event fsm.Event, args map[string]interface{}) error {
                fmt.Println("📦 Order created")
                args["created_at"] = time.Now()
                return nil
            },
        },
        {
            Name: OrderStatePending,
            OnEntry: func(ctx context.Context, from, to fsm.State, event fsm.Event, args map[string]interface{}) error {
                fmt.Println("⏳ Order pending payment")
                return nil
            },
        },
        {
            Name: OrderStatePaid,
            OnEntry: func(ctx context.Context, from, to fsm.State, event fsm.Event, args map[string]interface{}) error {
```

```python
# statemachine/core.py
"""
状态机核心实现
"""
from __future__ import annotations
import json
from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import (
    Any, Callable, Dict, Generic, List, Optional, Set, TypeVar,
    Protocol, runtime_checkable
)
import asyncio
from contextlib import asynccontextmanager

# 类型定义
State = str
Event = str
T = TypeVar('T')


# 守卫函数类型
GuardFunc = Callable[["TransitionContext"], Optional[str]]
# 动作函数类型
ActionFunc = Callable[["TransitionContext"], None]


@dataclass
class TransitionContext:
    """转换上下文"""
    instance: FSMInstance
    from_state: State
    to_state: State
    event: Event
    args: Dict[str, Any]

    def get_arg(self, key: str, default: Any = None) -> Any:
        """获取参数"""
        return self.args.get(key, default)

    def set_data(self, key: str, value: Any) -> None:
        """设置实例数据"""
        self.instance.data[key] = value

    def get_data(self, key: str, default: Any = None) -> Any:
        """获取实例数据"""
        return self.instance.data.get(key, default)


@dataclass
class Transition:
    """状态转换定义"""
    from_state: State
    to_state: State
    event: Event
    guard: Optional[GuardFunc] = None
    before: Optional[ActionFunc] = None
    after: Optional[ActionFunc] = None


@dataclass
class StateDefinition:
    """状态定义"""
    name: State
    on_entry: Optional[ActionFunc] = None
    on_exit: Optional[ActionFunc] = None
    is_final: bool = False


@dataclass
class TransitionRecord:
    """转换记录"""
    from_state: State
    to_state: State
    event: Event
    timestamp: datetime
    success: bool
    error: Optional[str] = None
    args: Dict[str, Any] = field(default_factory=dict)


@dataclass
class FSMInstance:
    """状态机实例"""
    id: str
    fsm_type: str
    current_state: State
    previous_state: Optional[State] = None
    data: Dict[str, Any] = field(default_factory=dict)
    history: List[TransitionRecord] = field(default_factory=list)
    created_at: datetime = field(default_factory=datetime.now)
    updated_at: datetime = field(default_factory=datetime.now)
    version: int = 1

    def to_dict(self) -> Dict[str, Any]:
        """转换为字典"""
        return {
            "id": self.id,
            "fsm_type": self.fsm_type,
            "current_state": self.current_state,
            "previous_state": self.previous_state,
            "data": self.data,
            "history": [
                {
                    "from_state": r.from_state,
                    "to_state": r.to_state,
                    "event": r.event,
                    "timestamp": r.timestamp.isoformat(),
                    "success": r.success,
                    "error": r.error,
                    "args": r.args,
                }
                for r in self.history
            ],
            "created_at": self.created_at.isoformat(),
            "updated_at": self.updated_at.isoformat(),
            "version": self.version,
        }

    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> FSMInstance:

        return cls(

                    from_state=r["from_state"],

class FSMConfig:
    """状态机配置"""

    def __init__(
        self,
        name: str,
        initial_state: State,
        states: List[StateDefinition],
        transitions: List[Transition],
        final_states: Optional[Set[State]] = None,
    ):
        self.name = name
        self.initial_state = initial_state
        self.states = {s.name: s for s in states}
        self.transitions: Dict[State, Dict[Event, Transition]] = {}
        self.final_states = final_states or set()

        # 构建转换索引
        for trans in transitions:
            if trans.from_state not in self.transitions:
                self.transitions[trans.from_state] = {}
            self.transitions[trans.from_state][trans.event] = trans


class StateMachine:
    """状态机"""

    def __init__(self, config: FSMConfig, persistence: Optional[Persistence] = None):
        self.config = config
        self.persistence = persistence
        self._listeners: List[Callable[[str, FSMInstance, Dict[str, Any]], None]] = []

    def add_listener(
        self,
        listener: Callable[[str, FSMInstance, Dict[str, Any]], None]
    ) -> StateMachine:
        """添加事件监听器"""
        self._listeners.append(listener)
        return self

    def _notify(
        self,
        event_type: str,
        instance: FSMInstance,
        data: Dict[str, Any]
    ) -> None:
        """通知监听器"""
        for listener in self._listeners:
            try:
                listener(event_type, instance, data)
            except Exception as e:
                print(f"Listener error: {e}")

    async def create_instance(
        self,
        instance_id: str,
        initial_data: Optional[Dict[str, Any]] = None
    ) -> FSMInstance:
        """创建状态机实例"""
        instance = FSMInstance(
            id=instance_id,
            fsm_type=self.config.name,
            current_state=self.config.initial_state,
            data=initial_data or {},
        )

        # 执行初始状态的进入动作
        initial_state = self.config.states.get(self.config.initial_state)
        if initial_state and initial_state.on_entry:
            ctx = TransitionContext(
                instance=instance,
                from_state="",
                to_state=self.config.initial_state,
                event="",
                args={},
            )
            initial_state.on_entry(ctx)

        # 持久化
        if self.persistence:
            await self.persistence.save(instance)

        self._notify("instance_created", instance, {})
        return instance

    async def trigger(
        self,
        instance: FSMInstance,
        event: Event,
        args: Optional[Dict[str, Any]] = None,
    ) -> bool:
        """
        触发状态转换

        Args:
            instance: 状态机实例
            event: 事件
            args: 额外参数

        Returns:
            转换是否成功
        """
        args = args or {}

        # 检查当前状态是否允许该事件
        state_transitions = self.config.transitions.get(instance.current_state)
        if not state_transitions:
            raise ValueError(f"No transitions defined for state {instance.current_state}")

        transition = state_transitions.get(event)
        if not transition:
            raise ValueError(
                f"Event '{event}' not allowed in state '{instance.current_state}'"
            )

        # 检查是否是终止状态
        if instance.current_state in self.config.final_states:
            raise ValueError(f"Cannot transition from final state {instance.current_state}")

        current_state_def = self.config.states.get(instance.current_state)
        target_state_def = self.config.states.get(transition.to_state)

        # 创建转换记录
        record = TransitionRecord(
            from_state=instance.current_state,
            to_state=transition.to_state,
            event=event,
            timestamp=datetime.now(),
            success=False,
            args=args,
        )

        # 创建转换上下文
        ctx = TransitionContext(
            instance=instance,
            from_state=instance.current_state,
            to_state=transition.to_state,
            event=event,
            args=args,
        )

        try:
            # 执行守卫条件
            if transition.guard:
                error_msg = transition.guard(ctx)
                if error_msg:
                    record.error = f"Guard failed: {error_msg}"
                    instance.history.append(record)
                    raise ValueError(record.error)

            # 执行转换前动作
            if transition.before:
                transition.before(ctx)

            # 执行当前状态的退出动作
            if current_state_def and current_state_def.on_exit:
                current_state_def.on_exit(ctx)

            # 执行状态转换
            instance.previous_state = instance.current_state
            instance.current_state = transition.to_state
            instance.updated_at = datetime.now()
            instance.version += 1

            # 合并参数到数据
            instance.data.update(args)

            # 执行目标状态的进入动作
            if target_state_def and target_state_def.on_entry:
                target_state_def.on_entry(ctx)

            # 执行转换后动作
            if transition.after:
                transition.after(ctx)

            record.success = True
            instance.history.append(record)

            # 持久化
            if self.persistence:
                success = await self.persistence.update(instance)
                if not success:
                    raise RuntimeError("Optimistic lock failed")

            self._notify("state_changed", instance, {
                "from": record.from_state,
                "to": record.to_state,
                "event": event,
            })

            return True

        except Exception as e:
            record.error = str(e)
            instance.history.append(record)

            if self.persistence:
                await self.persistence.update(instance)

            self._notify("transition_failed", instance, {
                "event": event,
                "error": str(e),
            })

            raise

    def can_trigger(self, instance: FSMInstance, event: Event) -> bool:
        """检查是否可以触发指定事件"""
        state_transitions = self.config.transitions.get(instance.current_state)
        if not state_transitions:
            return False
        return event in state_transitions

    def get_available_events(self, instance: FSMInstance) -> List[Event]:
        """获取当前状态下可用的事件列表"""
        state_transitions = self.config.transitions.get(instance.current_state, {})
        return list(state_transitions.keys())

    def is_in_final_state(self, instance: FSMInstance) -> bool:
        """检查是否处于终止状态"""
        return instance.current_state in self.config.final_states

    def get_state_definition(self, state: State) -> Optional[StateDefinition]:
        """获取状态定义"""
        return self.config.states.get(state)

    @asynccontextmanager
    async def transaction(self, instance: FSMInstance):
        """事务上下文管理器"""
        # 保存当前状态以便回滚
        saved_state = instance.current_state
        saved_data = instance.data.copy()
        saved_history = instance.history.copy()

        try:
            yield
        except Exception:
            # 回滚
            instance.current_state = saved_state
            instance.data = saved_data
            instance.history = saved_history
            raise
```

```python
# statemachine/persistence.py
"""
持久化实现
"""
import json
import sqlite3
from contextlib import contextmanager
from pathlib import Path
from typing import List, Optional
import aiosqlite

from .core import Persistence, FSMInstance


class MemoryPersistence:
    """内存持久化（用于测试）"""

    def __init__(self):
        self._instances: dict[str, FSMInstance] = {}

    async def save(self, instance: FSMInstance) -> None:
        """保存实例"""
        import copy
        self._instances[instance.id] = copy.deepcopy(instance)

    async def load(self, instance_id: str) -> Optional[FSMInstance]:
        """加载实例"""
        import copy
        instance = self._instances.get(instance_id)
        return copy.deepcopy(instance) if instance else None

    async def update(self, instance: FSMInstance) -> bool:
        """更新实例"""
        existing = self._instances.get(instance.id)
        if existing and existing.version != instance.version - 1:
            return False
        await self.save(instance)
        return True

    async def delete(self, instance_id: str) -> None:
        """删除实例"""
        self._instances.pop(instance_id, None)

    async def list(
        self,
        fsm_type: Optional[str] = None,
        state: Optional[str] = None
    ) -> List[FSMInstance]:
        """列出实例"""
        import copy
        results = []
        for instance in self._instances.values():
            if fsm_type and instance.fsm_type != fsm_type:
                continue
            if state and instance.current_state != state:
                continue
            results.append(copy.deepcopy(instance))
        return results


class SQLitePersistence:
    """SQLite 持久化"""

    def __init__(self, db_path: str = "fsm.db"):
        self.db_path = db_path

    async def _init_db(self):
        """初始化数据库"""
        async with aiosqlite.connect(self.db_path) as db:
            await db.execute("""
                CREATE TABLE IF NOT EXISTS fsm_instances (
                    id TEXT PRIMARY KEY,
                    fsm_type TEXT NOT NULL,
                    current_state TEXT NOT NULL,
                    previous_state TEXT,
                    data TEXT,
                    history TEXT,
                    created_at TEXT NOT NULL,
                    updated_at TEXT NOT NULL,
                    version INTEGER NOT NULL DEFAULT 1
                )
            """)
            await db.execute(
                "CREATE INDEX IF NOT EXISTS idx_fsm_type ON fsm_instances(fsm_type)"
            )
            await db.execute(
                "CREATE INDEX IF NOT EXISTS idx_current_state ON fsm_instances(current_state)"
            )
            await db.commit()

    async def save(self, instance: FSMInstance) -> None:
        """保存实例"""
        await self._init_db()
        async with aiosqlite.connect(self.db_path) as db:
            await db.execute(
                """
                INSERT OR REPLACE INTO fsm_instances
                (id, fsm_type, current_state, previous_state, data, history,
                 created_at, updated_at, version)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
                """,
                (
                    instance.id,
                    instance.fsm_type,
                    instance.current_state,
                    instance.previous_state,
                    json.dumps(instance.data),
                    json.dumps([r.__dict__ for r in instance.history]),
                    instance.created_at.isoformat(),
                    instance.updated_at.isoformat(),
                    instance.version,
                ),
            )
            await db.commit()

    async def load(self, instance_id: str) -> Optional[FSMInstance]:
        """加载实例"""
        await self._init_db()
        async with aiosqlite.connect(self.db_path) as db:
            db.row_factory = aiosqlite.Row
            async with db.execute(
                "SELECT * FROM fsm_instances WHERE id = ?", (instance_id,)
            ) as cursor:
                row = await cursor.fetchone()
                if not row:
                    return None

                return FSMInstance(
                    id=row["id"],
                    fsm_type=row["fsm_type"],
                    current_state=row["current_state"],
                    previous_state=row["previous_state"],
                    data=json.loads(row["data"]) if row["data"] else {},
                    history=[
                        TransitionRecord(**r)
                        for r in json.loads(row["history"] or "[]")
                    ],
                    created_at=datetime.fromisoformat(row["created_at"]),
                    updated_at=datetime.fromisoformat(row["updated_at"]),
                    version=row["version"],
                )

    async def update(self, instance: FSMInstance) -> bool:
        """更新实例（乐观锁）"""
        await self._init_db()
        async with aiosqlite.connect(self.db_path) as db:
            cursor = await db.execute(
                """
                UPDATE fsm_instances SET
                    current_state = ?,
                    previous_state = ?,
                    data = ?,
                    history = ?,
                    updated_at = ?,
                    version = version + 1
                WHERE id = ? AND version = ?
                """,
                (
                    instance.current_state,
                    instance.previous_state,
                    json.dumps(instance.data),
                    json.dumps([r.__dict__ for r in instance.history]),
                    datetime.now().isoformat(),
                    instance.id,
                    instance.version,
                ),
            )
            await db.commit()
            return cursor.rowcount > 0

    async def delete(self, instance_id: str) -> None:
        """删除实例"""
        async with aiosqlite.connect(self.db_path) as db:
            await db.execute(
                "DELETE FROM fsm_instances WHERE id = ?", (instance_id,)
            )
            await db.commit()

    async def list(
        self,
        fsm_type: Optional[str] = None,
        state: Optional[str] = None
    ) -> List[FSMInstance]:
        """列出实例"""
        await self._init_db()

        query = "SELECT * FROM fsm_instances WHERE 1=1"
        params = []

        if fsm_type:
            query += " AND fsm_type = ?"
            params.append(fsm_type)
        if state:
            query += " AND current_state = ?"
            params.append(state)

        async with aiosqlite.connect(self.db_path) as db:
            db.row_factory = aiosqlite.Row
            async with db.execute(query, params) as cursor:
                rows = await cursor.fetchall()
                return [
                    FSMInstance(
                        id=row["id"],
                        fsm_type=row["fsm_type"],
                        current_state=row["current_state"],
                        previous_state=row["previous_state"],
                        data=json.loads(row["data"]) if row["data"] else {},
                        history=[
                            TransitionRecord(**r)
                            for r in json.loads(row["history"] or "[]")
                        ],
                        created_at=datetime.fromisoformat(row["created_at"]),
                        updated_at=datetime.fromisoformat(row["updated_at"]),
                        version=row["version"],
                    )
                    for row in rows
                ]
```

```python
"""
订单状态机示例

import asyncio
from datetime import datetime, timedelta
from statemachine.core import (
    StateMachine, FSMConfig, StateDefinition, Transition, TransitionContext

from statemachine.persistence import MemoryPersistence, SQLitePersistence


# 定义状态和事件
class OrderState:
    CREATED = "CREATED"
    PENDING_PAYMENT = "PENDING_PAYMENT"
    PAID = "PAID"
    PREPARING = "PREPARING"
    SHIPPED = "SHIPPED"
    DELIVERED = "DELIVERED"
    CANCELLED = "CANCELLED"
    REFUNDED = "REFUNDED"


class OrderEvent:
    SUBMIT = "SUBMIT"
    PAY = "PAY"
    PREPARE = "PREPARE"
    SHIP = "SHIP"
    DELIVER = "DELIVER"
    CANCEL = "CANCEL"
    REFUND = "REFUND"
    TIMEOUT = "TIMEOUT"


def create_order_fsm_config() -> FSMConfig:
    """创建订单状态机配置"""

    # 状态定义
    states = [
        StateDefinition(
            name=OrderState.CREATED,
            on_entry=lambda ctx: print("📦 Order created"),
        ),
        StateDefinition(
            name=OrderState.PENDING_PAYMENT,
            on_entry=lambda ctx: print("⏳ Order pending payment"),
        ),
        StateDefinition(
            name=OrderState.PAID,
            on_entry=lambda ctx: print("💰 Payment received"),
        ),
        StateDefinition(
            name=OrderState.PREPARING,
            on_entry=lambda ctx: print("📋 Order is being prepared"),
        ),
        StateDefinition(
            name=OrderState.SHIPPED,
            on_entry=lambda ctx: print("🚚 Order shipped"),
        ),
        StateDefinition(
            name=OrderState.DELIVERED,
            is_final=True,
            on_entry=lambda ctx: print("✅ Order delivered successfully"),
        ),
        StateDefinition(
            name=OrderState.CANCELLED,
            is_final=True,
            on_entry=lambda ctx: print("❌ Order cancelled"),
        ),
        StateDefinition(
            name=OrderState.REFUNDED,
            is_final=True,
            on_entry=lambda ctx: print("💸 Order refunded"),
        ),
    ]

    # 转换定义
    transitions = [
        # 正常流程
        Transition(
            from_state=OrderState.CREATED,
            to_state=OrderState.PENDING_PAYMENT,
            event=OrderEvent.SUBMIT,
            guard=lambda ctx: None if ctx.get_arg("amount", 0) > 0 else "Amount must be positive",
            before=lambda ctx: print("📝 Validating order..."),
            after=lambda ctx: print(f"📧 Sending payment reminder to {ctx.get_data('customer_email')}"),
        ),
        Transition(
            from_state=OrderState.PENDING_PAYMENT,
            to_state=OrderState.PAID,
            event=OrderEvent.PAY,
            before=lambda ctx: print("💳 Processing payment..."),
        ),
        Transition(
            from_state=OrderState.PAID,
            to_state=OrderState.PREPARING,
            event=OrderEvent.PREPARE,
            after=lambda ctx: print("📦 Notifying warehouse to prepare order"),
        ),
        Transition(
            from_state=OrderState.PREPARING,
            to_state=OrderState.SHIPPED,
            event=OrderEvent.SHIP,
            before=lambda ctx: ctx.set_data("tracking_number", f"TRK{datetime.now().timestamp()}"),
        ),
        Transition(
            from_state=OrderState.SHIPPED,
            to_state=OrderState.DELIVERED,
            event=OrderEvent.DELIVER,
        ),

        # 取消流程
        Transition(
            from_state=OrderState.CREATED,
            to_state=OrderState.CANCELLED,
            event=OrderEvent.CANCEL,
        ),
        Transition(
            from_state=OrderState.PENDING_PAYMENT,
            to_state=OrderState.CANCELLED,
            event=OrderEvent.CANCEL,
        ),
        Transition(
            from_state=OrderState.PENDING_PAYMENT,
            to_state=OrderState.CANCELLED,
            event=OrderEvent.TIMEOUT,
            before=lambda ctx: print("⏰ Payment timeout, cancelling order"),
        ),

        # 退款流程
        Transition(
            from_state=OrderState.PAID,
            to_state=OrderState.REFUNDED,
            event=OrderEvent.REFUND,
            guard=lambda ctx: (
                None if datetime.now() - ctx.get_data("paid_at", datetime.now()) < timedelta(days=30)
                else "Refund period expired"
            ),
        ),
        Transition(
            from_state=OrderState.PREPARING,
            to_state=OrderState.REFUNDED,
            event=OrderEvent.REFUND,
            after=lambda ctx: print("🛑 Stopped preparation, processing refund..."),
        ),
    ]

    return FSMConfig(
        name="OrderFSM",
        initial_state=OrderState.CREATED,
        states=states,
        transitions=transitions,
        final_states={OrderState.DELIVERED, OrderState.CANCELLED, OrderState.REFUNDED},
    )


async def main():
    """主函数"""
    # 选择持久化方式
    # persistence = MemoryPersistence()
    persistence = SQLitePersistence("order_fsm.db")

    # 创建状态机
    config = create_order_fsm_config()
    fsm = StateMachine(config, persistence)

    # 添加监听器
    def on_event(event_type: str, instance, data: dict):
        if event_type == "state_changed":
            print(f"  → State changed: {data['from']} → {data['to']} ({data['event']})")
        elif event_type == "transition_failed":
            print(f"  ✗ Transition failed: {data['error']}")

    fsm.add_listener(on_event)

    # 创建订单实例
    instance = await fsm.create_instance(
        instance_id="order-123",
        initial_data={
            "customer_email": "customer@example.com",
            "amount": 199.99,
        },
    )

    print(f"\n🚀 Starting order workflow: {instance.id}")
    print(f"Current state: {instance.current_state}\n")

    # 执行状态转换
    workflow = [
        (OrderEvent.SUBMIT, {"amount": 199.99}),
        (OrderEvent.PAY, {"payment_method": "credit_card"}),
        (OrderEvent.PREPARE, {}),
        (OrderEvent.SHIP, {}),
        (OrderEvent.DELIVER, {}),
    ]

    for event, args in workflow:
        print(f"→ Triggering event: {event}")
        try:
            await fsm.trigger(instance, event, args)
            print(f"Current state: {instance.current_state}\n")
        except Exception as e:
            print(f"Error: {e}\n")
            break

    # 打印执行历史
    print("\n📊 Execution History:")
    for i, record in enumerate(instance.history, 1):
        status = "✓" if record.success else "✗"
        print(f"  {i}. {status} {record.timestamp.strftime('%H:%M:%S')}: "
              f"{record.from_state} → {record.to_state} ({record.event})")

    # 检查最终状态
    if fsm.is_in_final_state(instance):
        print(f"\n✅ Order reached final state: {instance.current_state}")

    # 测试守卫条件
    print("\n--- Testing Guard Conditions ---")

    # 尝试从已交付状态退款（应该失败）
    delivered_instance = await fsm.create_instance(
        instance_id="order-456",
        initial_data={
            "customer_email": "test@example.com",
            "amount": 100.0,
        },
    )

    # 快速走完流程到 DELIVERED
    for event in [OrderEvent.SUBMIT, OrderEvent.PAY, OrderEvent.PREPARE,
                  OrderEvent.SHIP, OrderEvent.DELIVER]:
        await fsm.trigger(delivered_instance, event, {"amount": 100.0} if event == OrderEvent.SUBMIT else {})

    print(f"\nTrying to refund delivered order...")
    try:
        await fsm.trigger(delivered_instance, OrderEvent.REFUND, {})
    except Exception as e:
        print(f"Expected error: {e}")


if __name__ == "__main__":
    asyncio.run(main())
```
