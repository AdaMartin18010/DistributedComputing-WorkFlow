## 一、核心概念

### 1.1 大数据平台定义

大数据平台是企业统一的数据基础设施，提供数据采集、存储、计算、分析、治理的全链路能力，支撑数据驱动的业务决策和智能化应用。

### 1.2 关键特性

- **海量存储**：PB级数据存储能力
- **弹性计算**：按需扩展的计算资源
- **实时处理**：毫秒级延迟的流处理能力
- **数据治理**：统一的数据标准和质量管理

### 1.3 适用场景

| 场景 | 适用性 | 说明 |
|------|--------|------|
| 实时报表 | ⭐⭐⭐⭐⭐ | 业务实时监控大屏 |

// 自定义分区器：按业务主键分区，保证同订单数据顺序
public class OrderPartitioner implements Partitioner {
    
    @Override
    public int partition(String topic, Object key, byte[] keyBytes,
                        Object value, byte[] valueBytes, Cluster cluster) {
        List<PartitionInfo> partitions = cluster.partitionsForTopic(topic);
        int numPartitions = partitions.size();
        
        // 订单ID哈希取模
        String orderId = (String) key;
        return Math.abs(orderId.hashCode()) % numPartitions;
    }
}

#### 3.3.2 预分区设计

```java
// 预分区避免热点
public class HBaseTableCreator {
    
    public void createOrderTable() throws IOException {
        Configuration conf = HBaseConfiguration.create();
        Connection conn = ConnectionFactory.createConnection(conf);
        Admin admin = conn.getAdmin();
        
        TableName tableName = TableName.valueOf("order_table");
        TableDescriptorBuilder builder = TableDescriptorBuilder.newBuilder(tableName);
        
        // 列族设置
        ColumnFamilyDescriptor infoCf = ColumnFamilyDescriptorBuilder
            .newBuilder(Bytes.toBytes("info"))
            .setCompressionType(Compression.Algorithm.SNAPPY)
            .setBloomFilterType(BloomType.ROW)
            .build();
        builder.setColumnFamily(infoCf);
        
        // 预分区：按hash分区
        byte[][] splitKeys = new byte[16][];
        for (int i = 0; i < 16; i++) {
            splitKeys[i] = Bytes.toBytes(String.format("%02x", i));
        }
        
        admin.createTable(builder.build(), splitKeys);
    }
}
```

### 4.2 Spark离线计算

#### 4.2.1 离线数仓ETL

```python
# Spark SQL ETL任务示例
from pyspark.sql import SparkSession
from pyspark.sql.functions import *

spark = SparkSession.builder \
    .appName("DailyETL") \
    .config("spark.sql.adaptive.enabled", "true") \
    .config("spark.sql.adaptive.coalescePartitions.enabled", "true") \
    .getOrCreate()

# 1. 读取ODS层数据
ods_order = spark.read.parquet("hdfs:///data/raw/db/mysql/order/")

# 2. 数据清洗转换
dwd_order = ods_order \
    .filter(col("is_deleted") == 0) \
    .withColumn("dt", date_format(col("create_time"), "yyyy-MM-dd")) \
    .withColumn("order_amount", col("amount").cast("decimal(18,2)")) \
    .select(
        "order_id",
        "user_id",
        "order_amount",
        "order_status",
        "create_time",
        "dt"
    )

# 3. 写入DWD层
dwd_order.write \
    .partitionBy("dt") \
    .mode("overwrite") \
    .parquet("hdfs:///data/cleaned/trade/order/")

# 4. DWS层聚合计算
dws_daily = dwd_order.groupBy("dt") \
    .agg(
        count("order_id").alias("order_cnt"),
        sum("order_amount").alias("order_amount"),
        countDistinct("user_id").alias("uv")
    )

dws_daily.write \
    .mode("overwrite") \
    .jdbc(jdbc_url, "dws_trade_daily", properties)
```

#### 4.3.2 Flink SQL实时计算

```sql
-- 创建Kafka源表
CREATE TABLE ods_order (
    order_id STRING,
    user_id STRING,
    amount DECIMAL(18,2),
    status STRING,
    create_time TIMESTAMP(3),
    proctime AS PROCTIME(),
    WATERMARK FOR create_time AS create_time - INTERVAL '5' SECOND
) WITH (
    'connector' = 'kafka',
    'topic' = 'ods_order_db_order',
    'properties.bootstrap.servers' = 'kafka:9092',
    'properties.group.id' = 'flink-order-consumer',
    'format' = 'json',
    'scan.startup.mode' = 'latest-offset'
);

-- 创建HBase维表
CREATE TABLE dim_user (
    user_id STRING,
    user_name STRING,
    user_level STRING,
    PRIMARY KEY (user_id) NOT ENFORCED
) WITH (
    'connector' = 'hbase-2.2',
    'table-name' = 'dim_user',
    'zookeeper.quorum' = 'zk:2181'
);

-- 创建结果表
CREATE TABLE dws_order_summary (
    window_start TIMESTAMP(3),
    window_end TIMESTAMP(3),
    user_level STRING,
    order_cnt BIGINT,
    order_amount DECIMAL(18,2),
    PRIMARY KEY (window_start, user_level) NOT ENFORCED
) WITH (
    'connector' = 'jdbc',
    'url' = 'jdbc:mysql://mysql:3306/analytics',
    'table-name' = 'dws_order_summary',
    'username' = 'root',
    'password' = 'password'
);

-- 实时聚合计算
INSERT INTO dws_order_summary
SELECT 
    TUMBLE_START(create_time, INTERVAL '1' MINUTE) as window_start,
    TUMBLE_END(create_time, INTERVAL '1' MINUTE) as window_end,
    u.user_level,
    COUNT(*) as order_cnt,
    SUM(o.amount) as order_amount
FROM ods_order o
LEFT JOIN dim_user FOR SYSTEM_TIME AS OF o.proctime u
    ON o.user_id = u.user_id
GROUP BY 
    TUMBLE(create_time, INTERVAL '1' MINUTE),
    u.user_level;
```

### 5.3 数据质量监控

#### 5.3.1 质量规则

```python
# 数据质量检测规则
QUALITY_RULES = {
    "completeness": {
        "user_id_not_null": "user_id IS NOT NULL",
        "order_amount_not_null": "order_amount IS NOT NULL"
    },
    "uniqueness": {
        "order_id_unique": "COUNT(DISTINCT order_id) = COUNT(*)"
    },
    "consistency": {
        "amount_positive": "order_amount > 0",
        "status_valid": "status IN ('PAID', 'UNPAID', 'REFUNDED')"
    },
    "timeliness": {
        "data_freshness": "MAX(create_time) >= NOW() - INTERVAL 1 HOUR"
    }
}

# 质量评分计算
def calculate_quality_score(table_name, check_results):
    total_rules = len(check_results)
    passed_rules = sum(1 for r in check_results if r.passed)
    return passed_rules / total_rules * 100
```
