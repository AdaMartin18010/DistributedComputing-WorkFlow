# 密钥分发与管理

## 概述

密钥是加密系统的核心，密钥的安全管理直接决定了整个安全体系的有效性。在分布式系统中，密钥管理面临更多挑战：如何在多个节点间安全地分发密钥？如何实现密钥轮换而不影响服务？如何应对密钥泄露事件？本文将系统介绍密钥生命周期管理的最佳实践。

## 二、密钥生成

### 2.1 安全密钥生成原则

```python
import secrets
import os
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
from cryptography.hazmat.primitives.kdf.argon2 import Argon2id

class SecureKeyGeneration:
    """
    安全密钥生成最佳实践
    """

    @staticmethod
    def generate_random_key(length_bytes: int = 32) -> bytes:
        """
        生成加密安全的随机密钥

        使用secrets模块（Python 3.6+），基于os.urandom()
        使用系统的CSPRNG（/dev/urandom, CryptGenRandom等）
        """
        # 方式1：secrets模块（推荐）
        key_secrets = secrets.token_bytes(length_bytes)

        # 方式2：os.urandom
        key_os = os.urandom(length_bytes)

        return key_secrets

    @staticmethod
    def generate_urlsafe_token(length: int = 32) -> str:
        """生成URL安全的随机令牌（用于API Key等）"""
        return secrets.token_urlsafe(length)

    @staticmethod
    def derive_key_from_password(password: str, salt: bytes = None,
                                  key_length: int = 32) -> tuple:
        """
        从密码派生密钥（使用Argon2id - 现代密码哈希冠军）

        Args:
            password: 用户密码
            salt: 可选盐值（不提供则生成新的）
            key_length: 派生密钥长度

        Returns:
            (派生密钥, 使用的盐值)
        """
        if salt is None:
            salt = os.urandom(16)

        # Argon2id 参数（OWASP推荐）
        # time_cost: 迭代次数
        # memory_cost: 内存使用(KB)
        # parallelism: 并行度
        kdf = Argon2id(
            salt=salt,
            length=key_length,
            iterations=3,
            lanes=4,
            memory_cost=65536  # 64MB
        )

        key = kdf.derive(password.encode())
        return key, salt

    @staticmethod
    def derive_key_hkdf(master_key: bytes,
                        context: bytes,
                        key_length: int = 32) -> bytes:
        """
        使用HKDF从主密钥派生子密钥

        适用于：从单个主密钥派生多个用途的密钥
        """
        from cryptography.hazmat.primitives.kdf.hkdf import HKDF

        hkdf = HKDF(
            algorithm=hashes.SHA256(),
            length=key_length,
            salt=None,  # 可选，增加安全性
            info=context,  # 区分不同用途的密钥
        )

        return hkdf.derive(master_key)


# 使用示例
if __name__ == "__main__":
    gen = SecureKeyGeneration()

    # 生成AES-256密钥
    aes_key = gen.generate_random_key(32)
    print(f"AES-256 Key (hex): {aes_key.hex()}")

    # 生成API密钥
    api_key = gen.generate_urlsafe_token(32)
    print(f"API Key: {api_key}")

    # 从密码派生密钥
    password = "user_password_123"
    derived_key, salt = gen.derive_key_from_password(password)
    print(f"\nDerived Key (hex): {derived_key.hex()}")
    print(f"Salt (hex): {salt.hex()}")

    # 从同一主密钥派生不同用途的密钥
    master = gen.generate_random_key(32)
    enc_key = gen.derive_key_hkdf(master, b"encryption-key-v1")
    auth_key = gen.derive_key_hkdf(master, b"authentication-key-v1")
    print(f"\nEncryption Key: {enc_key.hex()[:16]}...")
    print(f"Authentication Key: {auth_key.hex()[:16]}...")
```

### 3.2 密钥分发服务实现

```python
import base64
import json
import time
from cryptography.hazmat.primitives.ciphers.aead import AESGCM
from cryptography.hazmat.primitives import hashes, serialization
from cryptography.hazmat.primitives.asymmetric import rsa, padding
import redis

class KeyDistributionService:
    """
    分布式密钥分发服务
    使用信封加密 + 短期访问令牌
    """

    def __init__(self, redis_client: redis.Redis):
        self.redis = redis_client
        self.master_key = self._load_or_generate_master_key()

    def _load_or_generate_master_key(self) -> bytes:
        """加载或生成主密钥（实际应从HSM加载）"""
        # 这里简化为生成新密钥，生产环境应从安全存储加载
        return AESGCM.generate_key(bit_length=256)

    def request_data_key(self, service_id: str,
                         ttl_seconds: int = 3600) -> dict:
        """
        为服务请求数据加密密钥（DEK）

        Returns:
            {
                'key_id': 密钥标识,
                'encrypted_key': 用服务公钥加密的DEK,
                'expires_at': 过期时间
            }
        """
        # 生成新的DEK
        dek = AESGCM.generate_key(bit_length=256)
        key_id = f"dek:{service_id}:{int(time.time() * 1000)}"

        # 获取服务公钥
        service_public_key_pem = self._get_service_public_key(service_id)
        public_key = serialization.load_pem_public_key(service_public_key_pem)

        # 用服务公钥加密DEK（信封加密）
        encrypted_dek = public_key.encrypt(
            dek,
            padding.OAEP(
                mgf=padding.MGF1(algorithm=hashes.SHA256()),
                algorithm=hashes.SHA256(),
                label=None
            )
        )

        # 同时存储加密后的DEK在KMS中（用于密钥轮换等管理）
        encrypted_dek_for_storage = self._encrypt_with_master_key(dek, key_id)
        self.redis.setex(
            f"kms:key:{key_id}",
            ttl_seconds * 2,  # 存储略长一些
            encrypted_dek_for_storage
        )

        return {
            'key_id': key_id,
            'encrypted_key': base64.b64encode(encrypted_dek).decode(),
            'algorithm': 'AES-256-GCM',
            'expires_at': int(time.time()) + ttl_seconds
        }

    def decrypt_data_key(self, key_id: str,
                        encrypted_dek: bytes,
                        service_private_key) -> bytes:
        """
        服务使用私钥解密获取DEK
        （这个操作在服务端执行）
        """
        dek = service_private_key.decrypt(
            encrypted_dek,
            padding.OAEP(
                mgf=padding.MGF1(algorithm=hashes.SHA256()),
                algorithm=hashes.SHA256(),
                label=None
            )
        )
        return dek

    def _encrypt_with_master_key(self, data: bytes,
                                  context: str) -> bytes:
        """使用主密钥加密数据"""
        aesgcm = AESGCM(self.master_key)
        nonce = os.urandom(12)
        ciphertext = aesgcm.encrypt(nonce, data, context.encode())
        return nonce + ciphertext

    def rotate_key(self, key_id: str) -> dict:
        """
        轮换指定密钥
        1. 生成新密钥
        2. 标记旧密钥为"可解密不可加密"
        3. 重新加密使用该密钥的数据（可选）
        """
        # 解析key_id获取服务ID
        parts = key_id.split(':')
        service_id = parts[1] if len(parts) > 1 else 'unknown'

        # 创建新密钥
        new_key_info = self.request_data_key(service_id)

        # 标记旧密钥状态
        self.redis.hset(f"kms:key:{key_id}:meta",
                       mapping={
                           'status': 'deprecated',
                           'rotated_to': new_key_info['key_id'],
                           'rotated_at': int(time.time())
                       })

        return new_key_info


# 客户端SDK
class KeyServiceClient:
    """
    密钥服务客户端SDK
    用于服务本地解密和缓存DEK
    """

    def __init__(self, service_id: str, private_key_pem: bytes):
        self.service_id = service_id
        self.private_key = serialization.load_pem_private_key(
            private_key_pem, password=None
        )
        self._key_cache = {}  # 本地密钥缓存

    def get_data_key(self, kms_endpoint: str) -> bytes:
        """从KMS获取DEK并缓存"""
        import requests

        # 请求新密钥
        response = requests.post(
            f"{kms_endpoint}/keys/request",
            json={'service_id': self.service_id}
        )

        key_info = response.json()
        encrypted_dek = base64.b64decode(key_info['encrypted_key'])

        # 用私钥解密
        dek = self.private_key.decrypt(
            encrypted_dek,
            padding.OAEP(
                mgf=padding.MGF1(algorithm=hashes.SHA256()),
                algorithm=hashes.SHA256(),
                label=None
            )
        )

        # 缓存密钥
        self._key_cache[key_info['key_id']] = {
            'key': dek,
            'expires_at': key_info['expires_at']
        }

        return dek
```

### 4.2 本地密钥安全存储

```python
import os
import stat
import keyring
from cryptography.hazmat.primitives.ciphers.aead import AESGCM
from cryptography.hazmat.primitives.kdf.argon2 import Argon2id

class LocalKeyStorage:
    """
    本地密钥安全存储
    使用系统密钥环 + 文件权限控制
    """

    def __init__(self, app_name: str):
        self.app_name = app_name
        self.key_file = os.path.expanduser(f"~/.{app_name}/keys")
        os.makedirs(os.path.dirname(self.key_file), exist_ok=True)

    def store_key_securely(self, key_name: str, key_data: bytes):
        """
        安全存储密钥
        1. 尝试使用系统密钥环（keyring）
        2. 回退到文件存储（严格权限控制）
        """
        try:
            # 尝试使用系统密钥环
            keyring.set_password(self.app_name, key_name,
                                base64.b64encode(key_data).decode())
            print(f"Key '{key_name}' stored in system keyring")
        except Exception:
            # 回退到文件存储
            self._store_in_file(key_name, key_data)

    def _store_in_file(self, key_name: str, key_data: bytes):
        """使用文件权限控制存储"""
        # 生成包装密钥（使用用户密码派生）
        wrapper_key = self._get_wrapper_key()

        # 加密密钥
        aesgcm = AESGCM(wrapper_key)
        nonce = os.urandom(12)
        encrypted_key = aesgcm.encrypt(nonce, key_data, key_name.encode())

        # 存储到文件（只有所有者读写）
        temp_file = f"{self.key_file}.{key_name}.tmp"
        with open(temp_file, 'wb') as f:
            f.write(nonce + encrypted_key)

        # 设置严格权限（仅所有者可读写）
        os.chmod(temp_file, stat.S_IRUSR | stat.S_IWUSR)

        # 原子重命名
        final_file = f"{self.key_file}.{key_name}"
        os.rename(temp_file, final_file)

        print(f"Key '{key_name}' stored in file with restricted permissions")

    def _get_wrapper_key(self) -> bytes:
        """
        获取包装密钥（从用户输入或密钥环）
        这是解锁本地密钥存储的主密码
        """
        # 实际应用应该使用安全的方式获取用户密码
        # 这里简化处理
        password = os.environ.get('KEY_WRAPPER_PASSWORD')
        if not password:
            raise ValueError("KEY_WRAPPER_PASSWORD environment variable required")

        # 使用固定盐值（实际应存储在配置中）
        salt = b'fixed_salt_change_in_production'

        kdf = Argon2id(
            salt=salt,
            length=32,
            iterations=3,
            lanes=4,
            memory_cost=65536
        )
        return kdf.derive(password.encode())

    def retrieve_key(self, key_name: str) -> bytes:
        """检索密钥"""
        # 尝试从密钥环获取
        try:
            key_b64 = keyring.get_password(self.app_name, key_name)
            if key_b64:
                return base64.b64decode(key_b64)
        except Exception:
            pass

        # 从文件获取
        return self._retrieve_from_file(key_name)

    def _retrieve_from_file(self, key_name: str) -> bytes:
        """从文件检索并解密密钥"""
        file_path = f"{self.key_file}.{key_name}"

        with open(file_path, 'rb') as f:
            data = f.read()

        nonce = data[:12]
        encrypted_key = data[12:]

        wrapper_key = self._get_wrapper_key()
        aesgcm = AESGCM(wrapper_key)

        return aesgcm.decrypt(nonce, encrypted_key, key_name.encode())

class ContainerKeyManager:
    """
    容器环境密钥管理
    支持Docker Secrets和Kubernetes Secrets
    """

    @staticmethod
    def read_docker_secret(secret_name: str) -> str:

        """
        secret_path = f"/run/secrets/{secret_name}"
        try:
            with open(secret_path, 'r') as f:
                return f.read().strip()
        except FileNotFoundError:
            # 回退到环境变量（开发环境）
            return os.environ.get(secret_name.upper(), '')

    @staticmethod
    def read_kubernetes_secret(secret_name: str,
                                namespace: str = 'default') -> dict:
        """
        从Kubernetes Secret读取
        需要配置ServiceAccount权限
        """
        from kubernetes import client, config

        config.load_incluster_config()  # 在Pod内运行

        v1 = client.CoreV1Api()
        secret = v1.read_namespaced_secret(secret_name, namespace)

        # Kubernetes Secrets存储为base64编码
        result = {}
        for key, value in secret.data.items():
            result[key] = base64.b64decode(value).decode()

        return result
```

### 5.2 零停机轮换实现

```python
import time
import threading
from cryptography.hazmat.primitives.ciphers.aead import AESGCM

class ZeroDowntimeKeyRotation:
    """
    零停机密钥轮换实现
    支持读写分离的密钥版本管理
    """

    def __init__(self):
        self._keys = {}  # key_id -> {version: int, key: bytes, status: str}
        self._current_version = 0
        self._lock = threading.RLock()

    def rotate_key(self, key_id: str) -> int:
        """
        轮换密钥，生成新版本

        Returns:
            新版本号
        """
        with self._lock:
            if key_id not in self._keys:
                self._keys[key_id] = {}

            # 标记旧版本为"仅解密"
            for version, key_info in self._keys[key_id].items():
                if key_info['status'] == 'active':
                    key_info['status'] = 'decrypt_only'

            # 生成新版本
            new_version = max(self._keys[key_id].keys(), default=0) + 1
            new_key = AESGCM.generate_key(bit_length=256)

            self._keys[key_id][new_version] = {
                'key': new_key,
                'status': 'active',
                'created_at': time.time()
            }

            return new_version

    def encrypt(self, key_id: str, plaintext: bytes,
                associated_data: bytes = None) -> dict:
        """
        加密数据（总是使用最新版本的密钥）
        """
        with self._lock:
            if key_id not in self._keys:
                raise ValueError(f"Key {key_id} not found")

            # 找到活跃版本的密钥
            active_version = None
            active_key = None

            for version, key_info in self._keys[key_id].items():
                if key_info['status'] == 'active':
                    active_version = version
                    active_key = key_info['key']
                    break

            if not active_key:
                raise RuntimeError("No active key found")

            # 执行加密
            aesgcm = AESGCM(active_key)
            nonce = os.urandom(12)
            ciphertext = aesgcm.encrypt(nonce, plaintext, associated_data)

            return {
                'key_id': key_id,
                'key_version': active_version,
                'nonce': base64.b64encode(nonce).decode(),
                'ciphertext': base64.b64encode(ciphertext).decode(),
                'algorithm': 'AES-256-GCM'
            }

    def decrypt(self, encrypted_data: dict,
                associated_data: bytes = None) -> bytes:
        """
        解密数据（自动选择正确版本的密钥）
        """
        key_id = encrypted_data['key_id']
        version = encrypted_data['key_version']

        with self._lock:
            if key_id not in self._keys:
                raise ValueError(f"Key {key_id} not found")

            if version not in self._keys[key_id]:
                raise ValueError(f"Key version {version} not found")

            key_info = self._keys[key_id][version]
            key = key_info['key']

            # 执行解密
            aesgcm = AESGCM(key)
            nonce = base64.b64decode(encrypted_data['nonce'])
            ciphertext = base64.b64decode(encrypted_data['ciphertext'])

            return aesgcm.decrypt(nonce, ciphertext, associated_data)

    def retire_old_versions(self, key_id: str,
                           keep_versions: int = 3):
        """
        退役旧版本密钥（在确认所有数据已重新加密后）
        """
        with self._lock:
            if key_id not in self._keys:
                return

            versions = sorted(self._keys[key_id].keys(), reverse=True)

            for old_version in versions[keep_versions:]:
                key_info = self._keys[key_id][old_version]
                if key_info['status'] == 'decrypt_only':
                    # 安全清除密钥数据
                    key_info['key'] = b'\x00' * 32
                    del self._keys[key_id][old_version]
                    print(f"Retired key {key_id} version {old_version}")


# 使用示例
if __name__ == "__main__":
    rotation = ZeroDowntimeKeyRotation()

    # 初始化密钥
    rotation.rotate_key("data-encryption-key")

    # 加密一些数据
    data1 = rotation.encrypt("data-encryption-key", b"Sensitive data 1")
    print(f"Encrypted v{data1['key_version']}")

    # 轮换密钥
    new_version = rotation.rotate_key("data-encryption-key")
    print(f"Rotated to version {new_version}")

    # 加密新数据（使用新版本）
    data2 = rotation.encrypt("data-encryption-key", b"Sensitive data 2")
    print(f"Encrypted v{data2['key_version']}")

    # 仍然可以解密旧数据
    decrypted1 = rotation.decrypt(data1)
    print(f"Decrypted v{data1['key_version']}: {decrypted1}")

    # 解密新数据
    decrypted2 = rotation.decrypt(data2)
    print(f"Decrypted v{data2['key_version']}: {decrypted2}")
```

## 六、密钥审计与监控

```python
import logging
from dataclasses import dataclass
from datetime import datetime
from typing import Optional
import json

@dataclass
class KeyAuditEvent:
    """密钥审计事件"""
    timestamp: datetime
    event_type: str  # generate, use, rotate, delete, export
    key_id: str
    key_version: Optional[str]
    user_or_service: str
    action: str
    result: str  # success, failure
    details: dict

class KeyAuditLogger:
    """
    密钥审计日志
    记录所有密钥生命周期事件
    """

    def __init__(self, log_file: str = "/var/log/key-audit.log"):
        self.logger = logging.getLogger("key-audit")
        self.logger.setLevel(logging.INFO)

        # 防止日志被篡改，使用独立日志文件
        handler = logging.FileHandler(log_file)
        handler.setFormatter(logging.Formatter(
            '%(asctime)s - %(message)s'
        ))
        self.logger.addHandler(handler)

        # 同时发送给SIEM（安全信息和事件管理）
        self.siem_endpoint = None

    def log_event(self, event: KeyAuditEvent):
        """记录审计事件"""
        log_entry = {
            'timestamp': event.timestamp.isoformat(),
            'event_type': event.event_type,
            'key_id': self._hash_key_id(event.key_id),  # 隐私保护
            'key_version': event.key_version,
            'actor': event.user_or_service,
            'action': event.action,
            'result': event.result,
            'details': event.details
        }

        self.logger.info(json.dumps(log_entry))

        # 敏感操作告警
        if event.event_type in ['delete', 'export', 'unauthorized_access']:
            self._send_alert(event)

    def _hash_key_id(self, key_id: str) -> str:
        """哈希化密钥ID，保护敏感信息"""
        import hashlib
        return hashlib.sha256(key_id.encode()).hexdigest()[:16]

    def _send_alert(self, event: KeyAuditEvent):
        """发送安全告警"""
        # 集成PagerDuty、Slack、邮件等
        alert_message = f"""
        SECURITY ALERT: Sensitive Key Operation

        Event: {event.event_type}
        Key: {event.key_id}
        Actor: {event.user_or_service}
        Time: {event.timestamp}
        """
        print(alert_message)  # 实际应发送到告警系统


class KeyHealthMonitor:
    """
    密钥健康监控
    检测过期密钥、异常使用模式等
    """

    def __init__(self, key_store):
        self.key_store = key_store
        self.alerts = []

    def check_key_health(self) -> list:
        """执行健康检查"""
        issues = []

        for key_id, key_info in self.key_store.get_all_keys().items():
            # 检查过期
            if key_info.get('expires_at'):
                days_to_expire = (key_info['expires_at'] - datetime.now()).days
                if days_to_expire < 7:
                    issues.append({
                        'severity': 'high',
                        'key_id': key_id,
                        'issue': f'Key expires in {days_to_expire} days'
                    })

            # 检查长期未轮换
            last_rotated = key_info.get('last_rotated')
            if last_rotated:
                days_since_rotation = (datetime.now() - last_rotated).days
                max_age = key_info.get('max_age_days', 90)
                if days_since_rotation > max_age:
                    issues.append({
                        'severity': 'medium',
                        'key_id': key_id,
                        'issue': f'Key not rotated for {days_since_rotation} days'
                    })

            # 检查异常使用频率
            usage_stats = key_info.get('usage_stats', {})
            if usage_stats.get('last_hour', 0) > 10000:
                issues.append({
                    'severity': 'high',
                    'key_id': key_id,
                    'issue': 'Unusually high key usage detected'
                })

        return issues
```

## 七、最佳实践总结

### 7.1 密钥管理检查清单

- [ ] **生成**：使用CSPRNG，避免硬编码密钥
- [ ] **分发**：使用信封加密或DH密钥交换，避免明文传输
- [ ] **存储**：HSM > KMS > 加密文件 > 环境变量
- [ ] **使用**：最小权限原则，审计所有访问
- [ ] **轮换**：定期自动轮换，支持紧急轮换
- [ ] **销毁**：安全擦除，不可恢复
- [ ] **监控**：完整的审计日志，异常检测告警

### 7.2 常见错误避免

```python
# ❌ 错误：硬编码密钥
API_KEY = "sk_live_abcdef123456"

# ✅ 正确：从环境变量或KMS获取
API_KEY = os.environ.get('API_KEY')

# ❌ 错误：使用MD5/SHA1派生密钥
digest = hashlib.md5(password).digest()

# ✅ 正确：使用Argon2id或PBKDF2
key = Argon2id(...).derive(password)

# ❌ 错误：重复使用nonce
nonce = b'fixed_nonce'

# ✅ 正确：每次加密生成新nonce
nonce = os.urandom(12)

# ❌ 错误：明文存储密钥
with open('key.txt', 'w') as f:
    f.write(key.hex())

# ✅ 正确：使用密钥环或HSM
keyring.set_password('app', 'key', key)
```
