## 概述

网络优化是分布式系统性能调优的关键领域，涵盖TCP协议栈调优、RDMA高性能网络、内核参数优化和零拷贝技术，可显著降低网络延迟并提升吞吐量。

## 核心概念

### TCP vs RDMA对比

| 特性 | TCP/IP | RDMA |
|------|--------|------|
| 协议层次 | 传输层+网络层 | 直接访问远程内存 |
| CPU参与度 | 高（数据拷贝、协议处理） | 低（绕过OS内核） |
| 延迟 | 10-100μs | <1μs |
| 吞吐量 | 受协议栈限制 | 接近网卡线速 |
| 硬件要求 | 普通网卡 | 支持RDMA的网卡 |
| 适用场景 | 通用网络通信 | 高性能计算、存储 |

## 技术细节

### TCP内核参数深度调优

```bash
# /etc/sysctl.conf - TCP深度优化配置

# ==================== 连接建立优化 ====================
net.ipv4.tcp_tw_reuse = 1          # 复用TIME_WAIT连接（仅客户端）
net.ipv4.tcp_tw_recycle = 0        # 禁用（内核4.12+已移除）
net.ipv4.tcp_fin_timeout = 30      # FIN_WAIT_2超时时间
net.ipv4.tcp_keepalive_time = 300  # Keepalive探测间隔
net.ipv4.tcp_keepalive_probes = 5  # Keepalive探测次数
net.ipv4.tcp_keepalive_intvl = 15  # Keepalive探测间隔

# ==================== 缓冲区优化 ====================
# 内核网络缓冲区
net.core.rmem_max = 134217728      # 接收缓冲区最大值(128MB)
net.core.wmem_max = 134217728      # 发送缓冲区最大值(128MB)
net.core.rmem_default = 262144     # 接收缓冲区默认值(256KB)
net.core.wmem_default = 262144     # 发送缓冲区默认值(256KB)
net.core.netdev_max_backlog = 50000  # 网卡队列包数量

# TCP缓冲区自动调优
net.ipv4.tcp_rmem = 4096 87380 134217728   # min default max
net.ipv4.tcp_wmem = 4096 65536 134217728   # min default max
net.ipv4.tcp_moderate_rcvbuf = 1   # 启用接收缓冲区自动调优
net.ipv4.tcp_adv_win_scale = 1     # 窗口扩大因子

# ==================== 拥塞控制优化 ====================
net.ipv4.tcp_congestion_control = bbr  # 使用BBR算法
net.ipv4.tcp_notsent_lowat = 16384     # 减少队列延迟
net.ipv4.tcp_slow_start_after_idle = 0 # 禁用空闲后的慢启动

# ==================== 连接追踪优化 ====================
net.netfilter.nf_conntrack_max = 1000000
net.netfilter.nf_conntrack_tcp_timeout_established = 600
net.netfilter.nf_conntrack_tcp_timeout_time_wait = 30

# 应用配置
sysctl -p
```

### BBR拥塞控制算法

```bash
# 启用BBR（需要Linux 4.9+）
# 1. 加载tcp_bbr模块
modprobe tcp_bbr

# 2. 配置sysctl
echo "net.ipv4.tcp_congestion_control=bbr" >> /etc/sysctl.conf
sysctl -p

# 3. 验证
sysctl net.ipv4.tcp_congestion_control
# 输出: net.ipv4.tcp_congestion_control = bbr
```

**BBR优势**：

- 基于带宽和RTT建模，而非丢包
- 更高的吞吐量和更低的延迟
- 对随机丢包不敏感

### RDMA编程模型

```c
// RDMA编程基础示例（InfiniBand/RDMA verbs）

#include <infiniband/verbs.h>

// 1. 打开设备
struct ibv_device **dev_list = ibv_get_device_list(NULL);
struct ibv_context *ctx = ibv_open_device(dev_list[0]);

// 2. 创建Protection Domain
struct ibv_pd *pd = ibv_alloc_pd(ctx);

// 3. 注册内存区域
char *buf = malloc(BUF_SIZE);
struct ibv_mr *mr = ibv_reg_mr(pd, buf, BUF_SIZE,
    IBV_ACCESS_LOCAL_WRITE | IBV_ACCESS_REMOTE_READ | IBV_ACCESS_REMOTE_WRITE);

// 4. 创建Completion Queue
struct ibv_cq *cq = ibv_create_cq(ctx, 100, NULL, NULL, 0);

// 5. 创建Queue Pair
struct ibv_qp_init_attr qp_init_attr = {
    .send_cq = cq,
    .recv_cq = cq,
    .qp_type = IBV_QPT_RC,  // Reliable Connection
    .cap = {
        .max_send_wr = 100,
        .max_recv_wr = 100,
        .max_send_sge = 1,
        .max_recv_sge = 1
    }
};
struct ibv_qp *qp = ibv_create_qp(pd, &qp_init_attr);

// 6. RDMA Write操作
struct ibv_sge sge = {
    .addr = (uint64_t)buf,
    .length = data_len,
    .lkey = mr->lkey
};

struct ibv_send_wr wr = {
    .wr_id = 1,
    .opcode = IBV_WR_RDMA_WRITE,  // RDMA写操作
    .sg_list = &sge,
    .num_sge = 1,
    .wr.rdma.remote_addr = remote_addr,
    .wr.rdma.rkey = remote_rkey
};

ibv_post_send(qp, &wr, NULL);
```

### RoCEv2配置（RDMA over Converged Ethernet）

```bash
# 1. 配置网卡PFC（Priority Flow Control）
mlnx_qos -i eth0 --pfc 0,0,0,1,0,0,0,0

# 2. 启用DCQCN（Datacenter Congestion Control）
ecmp_roce_enable=1

# 3. 配置DCTCP（Data Center TCP）
echo dctcp > /sys/module/tcp_cubic/parameters/tcp_congestion_control

# 4. 调整MTU（建议9000 for Jumbo Frames）
ip link set dev eth0 mtu 9000
```

### 零拷贝技术

```java
// Java NIO零拷贝示例
public class ZeroCopyServer {

    public void sendFile(String filePath, SocketChannel socketChannel)
            throws IOException {
        FileChannel fileChannel = new FileInputStream(filePath).getChannel();

        // 零拷贝：FileChannel -> SocketChannel
        // 数据不经过用户空间，直接从内核缓冲区发送到网络
        long transferred = fileChannel.transferTo(
            0,                    // position
            fileChannel.size(),   // count
            socketChannel         // target
        );

        System.out.println("Transferred: " + transferred + " bytes");
        fileChannel.close();
    }
}
```

```c

int socket_fd = socket(AF_INET, SOCK_STREAM, 0);
off_t offset = 0;
sendfile(socket_fd, file_fd, &offset, file_size);
```

### DPDK高性能数据包处理

```c
// DPDK初始化示例
#include <rte_eal.h>
#include <rte_ethdev.h>

int main(int argc, char **argv) {
    // 初始化EAL
    int ret = rte_eal_init(argc, argv);

    // 配置网卡
    uint16_t port_id = 0;
    struct rte_eth_conf port_conf = {
        .rxmode = {
            .max_rx_pkt_len = RTE_ETHER_MAX_LEN,
        }
    };

    rte_eth_dev_configure(port_id, 1, 1, &port_conf);

    // 分配内存池
    struct rte_mempool *mbuf_pool = rte_pktmbuf_pool_create(
        "MBUF_POOL", 8192, 250, 0,
        RTE_MBUF_DEFAULT_BUF_SIZE, rte_socket_id()
    );

    // 设置接收队列
    rte_eth_rx_queue_setup(port_id, 0, 128,
        rte_eth_dev_socket_id(port_id), NULL, mbuf_pool);

    // 启动网卡
    rte_eth_dev_start(port_id);

    // 收包循环
    struct rte_mbuf *bufs[32];
    while (1) {
        uint16_t nb_rx = rte_eth_rx_burst(port_id, 0, bufs, 32);
        for (int i = 0; i < nb_rx; i++) {
            // 处理数据包
            process_packet(bufs[i]);
            rte_pktmbuf_free(bufs[i]);
        }
    }
}
```

### 网络优化检查清单

```
□ TCP优化
  □ 启用tcp_tw_reuse
  □ 调整缓冲区大小适配BDP

  □ 调整Keepalive参数

  □ 优化网络中断亲和性

□ 应用优化
  □ 使用连接池
  □ 启用TCP Fast Open

  □ 启用RDMA/RoCE
  □ 使用DPDK
  □ 配置巨帧（Jumbo Frames）
```

### BDP（Bandwidth-Delay Product）计算

```
BDP = 带宽 × 延迟

示例：
- 带宽：1Gbps = 125MB/s
- RTT：10ms = 0.01s
- BDP = 125MB/s × 0.01s = 1.25MB

TCP缓冲区应至少等于BDP才能充分利用带宽：
net.core.rmem_max = 1250000
net.core.wmem_max = 1250000
```

### 中断亲和性优化

```bash
# 查看网卡中断
watch -n1 'cat /proc/interrupts | grep eth0'

# 将网卡中断绑定到特定CPU
echo "f" > /proc/irq/101/smp_affinity  # 绑定到CPU 0-3

# 启用RPS（Receive Packet Steering）
echo f > /sys/class/net/eth0/queues/rx-0/rps_cpus

# 启用RFS（Receive Flow Steering）
echo 32768 > /proc/sys/net/core/rps_sock_flow_entries
echo 4096 > /sys/class/net/eth0/queues/rx-0/rps_flow_cnt
```

### 常见问题

**Q: 如何确定TCP缓冲区大小？**

```bash
# 计算BDP并设置缓冲区
BDP=$(echo "125000000 * 0.01" | bc)  # 1Gbps * 10ms
sysctl -w net.core.rmem_max=$BDP
sysctl -w net.core.wmem_max=$BDP
```

**Q: RDMA连接建立失败？**

- 检查网卡固件版本
- 验证子网管理器（SM）运行状态
- 确认PFC/ECN配置正确

**Q: 网络吞吐量不达预期？**

- 使用iperf3测试基础带宽
- 检查是否有丢包（ifconfig查看errors/dropped）
- 分析TCP窗口是否成为瓶颈
