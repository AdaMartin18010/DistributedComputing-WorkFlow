# 零信任架构

## 概述

零信任（Zero Trust）是一种安全架构理念，核心原则是"永不信任，始终验证"。与传统边界安全模型不同，零信任假设网络内外部都存在威胁，要求对每个访问请求进行严格的身份验证和授权，无论请求来自何处。

### 2.2 身份与访问管理

```python
from dataclasses import dataclass
from typing import List, Dict, Optional
from enum import Enum
import time
import jwt

class TrustLevel(Enum):
    """信任级别"""
    UNTRUSTED = 0
    LOW = 1
    MEDIUM = 2
    HIGH = 3
    FULL = 4

class DevicePosture(Enum):
    """设备健康状态"""
    COMPLIANT = "compliant"
    NON_COMPLIANT = "non_compliant"
    HIGH_RISK = "high_risk"

@dataclass
class DeviceContext:
    """设备上下文信息"""
    device_id: str
    device_type: str  # laptop, mobile, tablet, server
    os_version: str
    patch_level: str
    encryption_enabled: bool
    antivirus_enabled: bool
    firewall_enabled: bool
    jailbroken: bool = False
    last_seen: float = 0
    
    def calculate_trust_score(self) -> float:
        """计算设备信任分数"""
        score = 100.0
        
        if not self.encryption_enabled:
            score -= 30
        if not self.antivirus_enabled:
            score -= 20
        if not self.firewall_enabled:
            score -= 10
        if self.jailbroken:
            score -= 50
        
        # 检查补丁时效性（简化）
        if self.patch_level < "2024-01":
            score -= 15
        
        return max(0, score)

@dataclass
class UserContext:
    """用户上下文信息"""
    user_id: str
    roles: List[str]
    mfa_verified: bool
    location: str
    login_time: float
    previous_login_time: Optional[float]
    failed_attempts: int
    
    def calculate_trust_score(self) -> float:
        """计算用户信任分数"""
        score = 100.0
        
        if not self.mfa_verified:
            score -= 40
        if self.failed_attempts > 3:
            score -= 30
        
        # 位置异常检测
        if self.previous_login_time:
            time_diff = self.login_time - self.previous_login_time
            # 短时间内跨地理位置登录
            if time_diff < 3600:  # 1小时内
                score -= 20
        
        return max(0, score)


class ZeroTrustPolicyEngine:
    """
    零信任策略决策引擎
    """
    
    def __init__(self):
        self.policies = []
        self.trust_threshold = 70.0
    
    def evaluate_access(self, 
                       user: UserContext,
                       device: DeviceContext,
                       resource: str,
                       action: str) -> Dict:
        """
        评估访问请求
        
        Returns:
            {
                'allowed': bool,
                'trust_level': TrustLevel,
                'conditions': list,  # 可能需要额外条件
                'session_ttl': int   # 会话有效期（秒）
            }
        """
        # 计算综合信任分数
        user_score = user.calculate_trust_score()
        device_score = device.calculate_trust_score()
        
        # 加权计算（可根据场景调整）
        combined_score = user_score * 0.5 + device_score * 0.5
        
        # 检查最低要求
        if device_score < 30:
            return {
                'allowed': False,
                'reason': 'Device does not meet minimum security requirements',
                'trust_level': TrustLevel.UNTRUSTED
            }
        
        # 基于分数决定访问级别
        if combined_score >= 90:
            trust_level = TrustLevel.FULL
            session_ttl = 3600  # 1小时
        elif combined_score >= 75:
            trust_level = TrustLevel.HIGH
            session_ttl = 1800  # 30分钟
        elif combined_score >= self.trust_threshold:
            trust_level = TrustLevel.MEDIUM
            session_ttl = 600   # 10分钟
            # 中等信任可能需要额外验证
            return {
                'allowed': True,
                'trust_level': trust_level,
                'conditions': ['step_up_auth'],
                'session_ttl': session_ttl
            }
        else:
            return {
                'allowed': False,
                'reason': 'Trust score below threshold',
                'trust_level': TrustLevel.LOW
            }
        
        return {
            'allowed': True,
            'trust_level': trust_level,
            'conditions': [],
            'session_ttl': session_ttl
        }
    
    def continuous_evaluation(self, 
                             session_token: str,
                             current_context: Dict) -> bool:
        """
        持续会话评估
        
        在会话期间定期检查上下文变化
        """
        # 解析会话token
        try:
            payload = jwt.decode(session_token, options={"verify_signature": False})
            initial_trust = payload.get('trust_level')
            issued_at = payload.get('iat')
            
            # 检查会话是否过期
            if time.time() - issued_at > payload.get('ttl', 3600):
                return False
            
            # 检查上下文变化
            if current_context.get('device_posture') == DevicePosture.HIGH_RISK:
                return False
            
            if current_context.get('location_changed'):
                # 位置突然变化，需要重新认证
                return False
            
            return True
            
        except jwt.InvalidTokenError:
            return False
```

### 3.2 零信任网络分段

```python
class NetworkSegmentation:
    """
    零信任网络分段实现
    使用微分段限制东西向流量
    """
    
    def __init__(self):
        self.segments = {}
        self.policies = []
    
    def define_segment(self, segment_id: str, 
                       workload_selector: dict) -> dict:
        """
        定义网络分段
        
        Args:
            segment_id: 分段标识
            workload_selector: 工作负载选择器（标签）
        """
        segment = {
            'id': segment_id,
            'selector': workload_selector,
            'allowed_ingress': [],
            'allowed_egress': []
        }
        self.segments[segment_id] = segment
        return segment
    
    def create_microsegment_policy(self, 
                                   source_segment: str,
                                   dest_segment: str,
                                   allowed_ports: list,
                                   allowed_protocols: list) -> dict:
        """
        创建微分段策略
        
        默认拒绝所有流量，只显式允许指定通信
        """
        policy = {
            'source': source_segment,
            'destination': dest_segment,
            'ports': allowed_ports,
            'protocols': allowed_protocols,
            'action': 'ALLOW',
            'log': True  # 记录所有允许的流量
        }
        self.policies.append(policy)
        return policy
    
    def generate_network_policy(self) -> list:
        """
        生成Kubernetes NetworkPolicy
        
        默认拒绝所有入站和出站流量
        """
        policies = []
        
        # 默认拒绝策略
        default_deny = {
            'apiVersion': 'networking.k8s.io/v1',
            'kind': 'NetworkPolicy',
            'metadata': {
                'name': 'default-deny-all',
                'namespace': 'production'
            },
            'spec': {
                'podSelector': {},
                'policyTypes': ['Ingress', 'Egress']
            }
        }
        policies.append(default_deny)
        
        # 为每个分段生成允许策略
        for policy in self.policies:
            source = self.segments[policy['source']]
            dest = self.segments[policy['destination']]
            
            netpol = {
                'apiVersion': 'networking.k8s.io/v1',
                'kind': 'NetworkPolicy',
                'metadata': {
                    'name': f"allow-{policy['source']}-to-{policy['destination']}",
                    'namespace': 'production'
                },
                'spec': {
                    'podSelector': dest['selector'],
                    'policyTypes': ['Ingress'],
                    'ingress': [{
                        'from': [{
                            'podSelector': source['selector']
                        }],
                        'ports': [
                            {'protocol': 'TCP', 'port': port}
                            for port in policy['ports']
                        ]
                    }]
                }
            }
            policies.append(netpol)
        
        return policies


# 使用示例
segmentation = NetworkSegmentation()

# 定义分段
segmentation.define_segment('web', {'tier': 'frontend'})
segmentation.define_segment('api', {'tier': 'backend'})
segmentation.define_segment('database', {'tier': 'database'})

# 定义策略
segmentation.create_microsegment_policy(
    'web', 'api', [8080], ['TCP']
)
segmentation.create_microsegment_policy(
    'api', 'database', [5432, 6379], ['TCP']
)
# 注意：web不能直接访问database
```

### 4.1 行为分析与异常检测

```python
import numpy as np
from collections import defaultdict
import time

class BehaviorAnalytics:
    """
    用户和实体行为分析（UEBA）
    检测异常行为
    """
    
    def __init__(self):
        self.user_baselines = {}
        self.event_history = defaultdict(list)
        self.anomaly_threshold = 2.5  # 标准差倍数
    
    def establish_baseline(self, user_id: str, 
                          historical_events: list):
        """
        为用户建立行为基线
        """
        # 分析历史行为模式
        access_times = [e['hour'] for e in historical_events]
        access_locations = [e['location'] for e in historical_events]
        access_resources = [e['resource'] for e in historical_events]
        
        baseline = {
            'typical_hours': {
                'mean': np.mean(access_times),
                'std': np.std(access_times)
            },
            'typical_locations': set(access_locations),
            'typical_resources': set(access_resources),
            'access_frequency': len(historical_events) / 30  # 日均访问
        }
        
        self.user_baselines[user_id] = baseline
    
    def analyze_event(self, user_id: str, event: dict) -> dict:
        """
        分析单个事件，检测异常
        
        Returns:
            {
                'risk_score': float,
                'anomalies': list,
                'recommendation': str
            }
        """
        baseline = self.user_baselines.get(user_id)
        if not baseline:
            return {'risk_score': 0.5, 'anomalies': [], 
                   'recommendation': 'baseline_not_established'}
        
        anomalies = []
        risk_score = 0.0
        
        # 检查访问时间
        event_hour = event.get('hour', 12)
        hour_mean = baseline['typical_hours']['mean']
        hour_std = baseline['typical_hours']['std']
        
        if hour_std > 0:
            hour_zscore = abs(event_hour - hour_mean) / hour_std
            if hour_zscore > self.anomaly_threshold:
                anomalies.append('unusual_access_time')
                risk_score += 0.3
        
        # 检查访问位置
        event_location = event.get('location')
        if event_location not in baseline['typical_locations']:
            anomalies.append('new_location')
            risk_score += 0.4
        
        # 检查资源访问
        event_resource = event.get('resource')
        if event_resource not in baseline['typical_resources']:
            anomalies.append('unusual_resource_access')
            risk_score += 0.2
        
        # 检查访问频率
        recent_events = self._get_recent_events(user_id, hours=1)
        if len(recent_events) > baseline['access_frequency'] * 3:
            anomalies.append('high_frequency_access')
            risk_score += 0.3
        
        recommendation = 'allow'
        if risk_score > 0.7:
            recommendation = 'block_and_alert'
        elif risk_score > 0.4:
            recommendation = 'step_up_auth'
        elif anomalies:
            recommendation = 'log_and_monitor'
        
        return {
            'risk_score': min(risk_score, 1.0),
            'anomalies': anomalies,
            'recommendation': recommendation
        }
    
    def _get_recent_events(self, user_id: str, hours: int) -> list:
        """获取最近的事件"""
        cutoff = time.time() - hours * 3600
        return [e for e in self.event_history[user_id] 
                if e['timestamp'] > cutoff]


class ContinuousVerification:
    """
    持续验证实现
    在会话期间持续评估信任状态
    """
    
    def __init__(self):
        self.active_sessions = {}
        self.verification_interval = 60  # 每60秒验证一次
    
    def start_session(self, user_id: str, 
                     device_id: str,
                     initial_context: dict) -> str:
        """启动会话"""
        session_id = f"{user_id}_{int(time.time())}"
        
        self.active_sessions[session_id] = {
            'user_id': user_id,
            'device_id': device_id,
            'context': initial_context,
            'trust_level': initial_context.get('trust_level'),
            'started_at': time.time(),
            'last_verified': time.time()
        }
        
        return session_id
    
    def verify_session(self, session_id: str,
                      current_context: dict) -> dict:
        """
        验证会话是否仍然有效
        
        在每次敏感操作前调用
        """
        session = self.active_sessions.get(session_id)
        if not session:
            return {'valid': False, 'reason': 'session_not_found'}
        
        # 检查会话过期
        session_ttl = 3600  # 1小时
        if time.time() - session['started_at'] > session_ttl:
            return {'valid': False, 'reason': 'session_expired'}
        
        # 设备状态变化检查
        if current_context.get('device_posture') == 'non_compliant':
            # 降低信任级别或终止会话
            return {
                'valid': True,
                'trust_level': 'reduced',
                'action': 'require_reauthentication'
            }
        
        # 位置突变检查
        last_location = session['context'].get('location')
        current_location = current_context.get('location')
        if last_location and current_location != last_location:
            # 检查位置变化是否合理
            time_elapsed = time.time() - session['last_verified']
            if not self._is_location_change_possible(
                last_location, current_location, time_elapsed):
                return {'valid': False, 'reason': 'impossible_travel'}
        
        # 更新会话状态
        session['last_verified'] = time.time()
        session['context'].update(current_context)
        
        return {'valid': True, 'trust_level': session['trust_level']}
```
